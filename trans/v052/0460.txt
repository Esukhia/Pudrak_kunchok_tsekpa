ཆ་བར་སྦྱོར་བ་ལ་བརྩོན་པ། དེ་དག་གི་སེམས་དང་སེམས་ལས་འབྱུང་བ་རྣམས་ཤེས་ཏེ། ཤེས་ནས་ཀྱང་དེ་དག་རྒྱད་བསྲིངས་ཏེ་ཡོངས་སུ་སྤོང་བར་འགྱུར་རོ། །འོད་སྲུང་དགེ་ཚུལ་སྐུལ་བྱེད་མི་
བཙང་བས་འབགས་པའི་ཕྱག་དར་ཁྲོད་པ་ཐོགས་ནས། བསོད་སོམས་བླངས་ཏེ་མཚོ་མ་དྲོས་པར་སོང་བ་ལ། དེ་དེར་སོང་བའི་ཚེ་མཚོ་མཆོག་དེ་ལ་གནས་པའི་ལྷ་དེ་དག་གིས་རྒྱང་
རིང་པོར་བསུ་སྟེ། རྐང་པ་ལ་བགོ་བོས་ཕྱག་འཚལ་ནས། གཙང་སྦྲ་སྤྱོད་པའི་ལྷ་དེ་དག་གིས་མི་གཙང་བས་འབགས་པའི་ཕྱག་དུར་ཁྲོད་པ་དེ་བདག་ཉིད་ཀྱིས་བཀྲུས་ཏེ། དྲི་མ་མེད་
པར་བྱས་ནས། བཀྲུས་པའི་ཆུ་དྲི་མ་ཅན་དེས་རང་གིས་ལུས་ལ་སྐུ་དེ། སྐུལ་བྱེད་ཀྱི་ཚུལ་ཁྲིམས་ཀྱི་ཡོན་ཏན་ཡང་དག་པར་བླངས་པ་དང་། སྙོམས་པར་ཞུགས་པ་ཤེས་ནས་མཚོ་དེའི་
ལྷ་རྣམས་ཡན་ལག་ལྔས་ཕྱག་འཚལ་ལོ། །ཀུན་ཏུ་རྒྱུ་རབ་ཏུ་བཟང་པོ་ཞེས་བྱས་བ་དེ་གོས་གཙང་མ་བགོས་ནས་བསོད་སྙོམས་བླངས་ཏེ། མཚོ་མ་དྲོས་པར་སོང་བ་དང་། མཚོ་དེ་ན་གནས་
པའི་ལྷ་དེ་དག་གིས །དེ་མཚོ་མཆོག་འདིར་ཁ་ཟས་ཟོས་ནས་ལྷག་མ་བོར་ན། མཚོ་མཆོག་མི་གཙང་བར་བྱས་སུ་འོང་ཞེས་རྒྱང་གྲགས་གཅིག་ཁོར་ཡུལ་ཙམ་གྱིས་བར་ནང་དུ་འདར་མ་སྟེར་བ་

པོ་ནི་འགྲོར་ཡང་མ་སྟེར་བ་ལ་སྟོས། འོད་སྲུང་དེ་ལྟ་བས་ན་དགེ་སློང་ཕྱག་དང་ཁྲོད་པ་པ་ཡོན་ཏན་གྱི་ཆོས་འདི་ལྟར་བུ་འདི་དག་འདོད་ལ། ཞུམ་ཞུམ་པར་མུ་གྱུར་པ་འཕགས་པའི་རིགས་ལ་གནས་པས་
