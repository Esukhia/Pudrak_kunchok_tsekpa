མ་ཡིན། ལུས་བརྒྱན་པའི་ཕྱིར་མ་ཡིན་ནོ། །འོད་སྲུང་འདི་ལ་མི་གཏི་མུག་པ་ཁ་ཅིག་ཁྱོད་དང་ངེའི་རྗེས་སུ་མི་བསློབ་ཅིང་རྐམ་པ་ཆགས་པའི་ཟིལ་ཀྱིས་ནོན་པ། ཆོས་གོས་སྩོགས་པར་བྱེད་པ། ལྷུང་ཟེད་སྩོགས་
པར་བྱེད་པ། འཇིག་རྟེན་གྱི་ཟང་ཟིང་སྩོགས་པར་བྱེད་པ། མགྲོན་བུ་དང་། གསེར་དང་། ནོར་དང་འབྲུ་སྩོགས་པར་བྱེད་པ། བ་ལང་དང་། ལུག་དང་། བྱ་གག་དང་། ཕག་དང་། བོང་བུ་དང་། རྟ་དང་། ཐོད
གཤོལ་དང་། ཤིང་རྟེ་འདྲེན་བ་ཉེ་བར་འཇོག་པར་བྱེད་པ། ཁྱིམ་པའི་ལས་ཀྱི་མཐའ་སྤྱོད་པ་དག་པས་ནི་འོད་སྲུང་ཁྱིམ་པ་དག་ཁྱད་བར་དུ་འགྱུར་བ་ཡོད་ཀྱི། མི་བླུན་པོ་རབ་ཏུ་འབྱུང་བའི་མཚམས་
ལས་འདས་པ་དེ་དག་ལ་ནི་ཁྱད་བར་དུ་འགྱུར་པ་གང་ཡང་མེད་དོ། །འོད་སྲུང་ཁྱིམ་པ་རྣམས་ཁྱད་བར་དུ་འགྱུར་བ་ནི་དགེ་སྦྱོང་མ་ཡིན་ལ། དགེ་སྦྱོང་དེ་ཁས་འཆེ་བ། ཚངས་པར་སྤྱོད་པ་མ་ཡིན་བ་ལ།
ཚངས་པར་སྤྱོད་པར་ཁས་འཆེ་བ། གོས་ངུར་སྨྲིག་བགོས་ཤིང་བྱ་བ་མང་བ། སྦག་པ་སྣ་ཚོགས་ཀྱིས་སྦགས་པ། ཟས་དང་གོས་ལྷུར་ལེན་པ་དེ་དག་ཁྱིམ་པ་དག་གིས་གོས་ངུར་སྨྲིག་གྱོན་པར་རིག་ནས། ཞེས་ས་དང་གུས་
པ་བསྐྱེད་པ་དང་། ཆོས་གོས་དང་བསོད་སྙོམས་ཀྱིས་སྡུད་པར་བྱེད་པ་དང་། གུས་པར་སྨྲས་ཞིང་ཕྱག་འཚལ་ལ། བར་བྱེད་པ་སྟེ། འདི་སྲུང་འདི་ནི་ཁྱིམ་པ་དེ་དག་གི་ཁྱད་བར་དུ་འགྱུར་བ་ཡིན་གྱི། དེ་དག་རབ་ཏུ་བྱུང་བ་དེ་དག་ལ་ནི་མེད་
པའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། འདི་ལྟར་དེ་དག་ལ་ནི་ཡོངས་སུ་ལེན་པ་ལ་བརྩོན་པ་མ་ཡིན་ཞིང་། ཅི་ཡང་མི་སྩོགས་པའི་ཕྱིར། འོད་སྲུང་དེའི་ཚེ་དེའི་དུས་ན་དགེ་སློང་ཕལ་ཆེར་ལྷུང་ཟེད་དང་། ཆོས་གོས་མང་ཞིང་སྣོད་སྤྱད་མང་བ་དང་། རྙེད་མང་
བར་འགྱུར་དེ། དེ་དག་ལ་སྟིང་སྟང་བྱེད་པར་འགྱུར། བཙུན་བར་བྱེད་པར་འགྱུར་། རི་མོ་བྱེད་པ་དང་མཆོད་པ་བྱེད་པར་འགྱུར་རོ། །ཅིའི་ཕྱིར་ཞེ་ན། འདི་ལྟར་དེ་སྦྱིན་པ་དང་། ཕྱིར་སྦྱིན་བ་དུས་དུས་སུ་བྱེད་ཅིང་བྱ་བ་ཇི་ལྟ་
