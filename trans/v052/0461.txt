འང་བྱ་ི་ྱི་བྱི་བི་བྱི་ྱི་ི་བྱི་ྱི་ི་ྱི་བ་ི་བྱ་ར་བུ་བརི་བརི་རྱིད་པ་རྱི་བི་རྱན་བརི་རི་རི་བ་བི་རིན་པི་ཤི་པི་སིས་རིས་རི་རིང་ོན་རིན་སྱོ་
༄༅། །མཆོད་རྟེན་གྱི་འདུ་ཤེས་དང་།སྟོན་པའི་འདུ་ཤེས་དང་། བདག་གིས་པ་མེད་ཅིང་ཡོངས་སུ་འཛིན་པ་མེད་པའི་འདུ་ཤེས་ཀྱིས་ལུས་ལ་ཕྱག་དར་ཁྲོད་པ་དག་བགོ་བར་བྱའོ། །དི་ལྟར་
ཡང་སེམས་ཀྱིས་ལུས་དག་པར་འགྱུར་གྱི་ལུས་ཀྱིས་སེམས་དག་པར་མི་འགྱུར་རོ། །དེ་བས་ན་སེམས་དག་པར་འགྱུར་བར་བྱའོ། །ལུས་ཚབས་དག་བ་ནི་མ་ཡིན་གྱི། སེམས་དག་པ་ནི་བཅོམ་ལྡན་འདས་ཀྱིས་བསྨན་
པ་ལ་ཚངས་པར་སྤྱོད་པ་ཞེས་གསུངས་སོ་སྙམ་དུ་སེམ་ངེས་པར་བསམ་པར་བྱའོ། འོད་སྲུང་དེ་ལྟར་ན་དགེ་སློང་ཕག་དངར་ ཁྲོད་པ་པས་ཁྱོད་དང་དེའི་རྗེས་སུ་བསླབ་བར་བྱའོ། །འོད་སྲུང་འདི་ལྟ་བུའི་དུས་དང་། འདི་ལྟ་
བུའི་མན་ཡོད་དེ། འོ་སྲུང་ཆོས་གོས་ངན་པ་གྱོན་པ། ཆོས་གོས་ངན་དོན་གྱིས་ཚོག་པར་འཛིན་པ་ཁྱོད་གང་དུ་ཆོས་གོས་སྔམ་སྤྱར་ཁྱེའུ་འམ་ཁྲེའུ་གི་སྟེང་ངུ་གཞག་སྟེ། ཁྱོད་བླ་གོས་གྱོན་ནས་ཆགས་ར་འཆག་
བའི་ཚོ་ལྷ་བརྒྱ་སྟོང་ཕྲག་མུང་པོ་དག་ལྷགས་ནས། འདི་ནི་ཚུལ་ཁྲིམ་དུང་ཏིང་འཛིན་དང་ཤེས་རབ་ཀྱི་མ་ཡོངས་སུ་བསྒོས་པའི་ལས་ལ་བགོ་བགོ་བའི་ཆོས་གོས་སོ་ཞེས་ཁྱོད་ཀྱིས་ཆོས་གོས་རྣམ་སྤྱར་དེ་ལ་ཕྱག་
འའཚལ་བར་འགྱུར་རོ། །དོས་སུ་ལྟོས་ཆོས་གོས་ལ་ཡང་ཕྱག་འཚལ་བར་འགྱུར་ན་ལས་ལ་ལྟ་ཅི་སོསོ། །འོད་སྲུང་ཇི་ལྟར་ན་དགེ་སློང་དེའི་རྗེས་སུ་སློབ་པ་དག་དེ་བཞིན་དུ་བསྒྲུབ་པར་འགྱུར་སྙམ་སྟེ། ངས་འཁོར་
ལོས་སྦྱུར་བའི་རྒྱལ་སྲིད་སྡང་ནས། རབ་ཏུ་བྱུང་སྟེ་ཀ་ཤི་པ་ཕྲ་མོ་དང་ད་གུ་ལེའི་རས་དང་། གོ་ཏམ་པའི་ངས་དང་། དང་རིས་ཅུན་གྱི་ཆོས་གོས་རྣམས་སྙངས་ནས་འཕགས་པའི་རིགས་ལ་དགའ་བའི་ཕྱི་ང་ཤ་ནའི་
ས་ཀྱིས་གོས་རྣམས་ལུས་ལ་བགོས་སོ། །འོད་སྲུངས་ཁྱོད་ལས་དར་གོས་ངན་པ་བླངས་ཏེ་དས་ཁྱོད་ལ་ཤ་ནའི་རས་བྱིན་བ་གང་ཡིན་པ་དེ་ཐམས་ཅད་ཀྱང་ཁྱོད་ཉིད་ལ་སྙིང་བརྩེ་བའི་ཕྱིར་གྱི་ཆགས་པའི་ཕྱིར་
