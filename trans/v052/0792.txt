བར་བྱའོ་ཞེས་སྒྲོགས་སོ། །དེ་བུད་མེད་དྲུག་ཁྲི་དང་ དགའ་དགུར་སྤྱོད་དེ། །དེ་དག་ཐམས་ཅད་ཀྱང་ལག་ན་མེ་ཏོག་ཐོགས་ན་ཤིན་ཏུ་དྲི་ཞིམ་པའི་ཆུ་བྱུང་ཅན། དུ་གུ་ལའི་རས་ཀྱི་གོས་བགོས་པ། །
བརྒྱན་དམ་པ་དང་། རྐང་གདུབ་སྨྲ་སྟན་པ། བཞིན་པད་མ་ལྟ་བུ་ལ་ཟླ་གམ་ལྟ་བུས་མཛས་པར་བྱས་པ་དག་སྦྲང་རྩི་དང༌། སྦྲང་ཆང་མངར་བ་དང་། མེ་ཏོག་གི་ཆང་གིས་རྒྱགས་ཤིང་མྱོས་པས་
ཁྱུད་ནས་འདུག་ཏེ། དེ་དག་ཐམས་ཅད་འདི་སྐད་ཅེས་བདག་ཅག་ཐམས་ཅད་ཇོ་བོ་འདིའི་གཡོག་དང་བསྙེན་བཀུར་བྱའོ། །ཁྱོད་འདི་ནི་བསོད་ནམས་ཀྱི་སྟོབས་ཀྱིས་བསྒྲུབས་པའི་མཐོ་རིས་ཀྱི་ཞིམ་
ཞིང་བདེ་བའི་ལྷའི་འདོད་པའི་ལོངས་སྤྱོད་ཤིས་པ་འདི་ལ་ཀུན་ཏུ་སྤྱོད་ཅིག་ཅེས་འཛེར་ཏོ། །དྲང་སྲོང་ཆེན་པོ་ལྷའི་བུ་དེ་དང་ལྷའི་བུ་མོ་དེ་དག་ལྷན་ཅིག་ཏུ་དགའ་ཚལ་དང་། འདུས་འཚལ་དང་།
ཨར་མོ་ནི་ག་ལྟ་བུའི་རྡོ་ལེབ་དང་། རྣམ་པར་མཉམ་འཕྲུལ་དང་། རབ་འབབས་དང་། སེང་སེར་པོ་དང་། མེ་ཏོག་གི་འཕྲལ་འཐབ་དང་། ཤིང་ཡོངས་འདུས་བསྟོལ་ལ་སྩོགས་པའི་སྐྱེད་མོས་
ཚལ་དམ་པ་མཆོག་ཏུ་བདེ་ཞིང་བསིལ་བ། གནོད་པའི་རླུང་མེད་པ་ལྷའི་བྱ་ནམ་མཁའ་ལ་འགྲོ་ཞེས་བྱ་བའི་སྒྲ་སྙན་པས་མཛེས་པར་བྱས་པ། ལྷའི་མེ་ཏོག་དང་། འབྲས་བུ་དང་། སྤོས་ཀྱིས་ཀུན་ཏུ་
གང་བ། བེ་དུརྱ་དང་། ནོར་བུ་དང་། སྦུག་དང་། རྡོའི་སྙིང་པོ་དང་། ས་སྣ་ཚོགས་ཀྱིས་སྒྲོན་མ་ཅན། བུང་བ་དང་། ཁུ་ལུང་རྒྱགས་པ་དང་། བྱ་སྒྲ་སྙན་པ་འབྱིན་པ། རྨ་བྱ་དང་། མདང་གཤོགས་
རྩེ་མོ་གསེར་ཅན་ཡོད་པ། ནོར་བུ་དང་རིན་པོ་ཆེའི་ཐིག་ལེས་རྣམ་པར་བཀྲ་ཤིང་བཀྲ་ཤིས་པ་ཡོད་པ། སྣ་མཐོ་ཞིང་རིང་བ། བེ་དུརྱའི་མཆུ་ལས་སྒྲ་སྐད་ཉན་པ་སྣ་ཚོགས་འཐུན་པར་སྒྲོགས་པ།།
