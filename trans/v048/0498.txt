ཅག་དེ་ཡང་ངས་ཕྱི་མའི་སྐྱེ་བོ་རྣམས་ལ་ཐུགས་བརྩེབས་ངའི་ཉན་ཐོས་རྒས་འཁོགས་ཏེ་གཏུགས་པ་གང་དག་གི་ལུས་ལུ་བསྙེན་བཀུར་བ་དགོས་པ་དག་འབྱུང་བར་འགྱུར་བ་དེ་དག་དེ་བཞིན་གཤེགས་པས་
རྗེས་སུ་མ་གནང་ངོ་ཞེས་འཕོངས་པར་མི་འགྱུར་བར་བྱ་བའི་ཕྱིར་དེ་སྐད་གསུངས་ཏེ། དེ་བཞིན་གཤེགས་པས་དེ་ལ་དགོངས་ནས་དེ་སྐད་གསུངས་པ་ཡིན་ན། དེ་ལ་མི་བླུན་པོ་ཁ་ཅིག་དེ་བཞིན་གཤེགས་
པ་བགྲེས་འཁོགས་ཏེ། གཏུགས་གོ་ཞེས་སྨྲ་བཞིན་ཁོ་ནར་འཛིན་པར་བྱེད་དེ། དེ་དག་ཀྱང་རང་ཡང་ཉམས་པར་བྱས་ལ་གཞན་ཡང་ཉམས་པར་བྱེད་པ་ཡིན་ནོ། །གསོལ་པ་བཅོམ་ལྡན་འདས་མེའུ་དགལ་
གྱི་བུ་ཁྱེད་སོང་ལ་སྨན་པའི་རྒྱལ་པོ་འཚོ་བྱེད་ཀྱི་གནས་ནས། སྨན་ལོང་ཤིག་དང་བཟའ་བར་བྱའོ་ཞེས་གསུངས་པ་དེ་ཇི་ལྟར་ལྟ་བར་བགྱི་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ་རིགས་
ཀྱི་བུ་དེ་ཡང་ངས་ཕྱི་མའི་སྐྱེ་བོ་རྣམས་ལ་ཐུགས་བཙོ་བས་ངའི་ཉན་ཐོས་གང་དག་ལ་སྨན་དགོས་པ་དག་འབྱུང་བར་འགྱུར་བ་དེ་དག་ལ། དེ་བཞིན་གཤེགས་པས་རྗེས་སུ་མ་གནང་། རྗེས་སུ་མ་གནང་
ངོ་ཞེས་ཕོངས་པར་མི་འགྱུར་བར་བྱ་བའི་ཕྱིར་དེ་སྐད་གསུངས་ཏེ། དེ་བཞིན་གཤེགས་པས་དེ་ལས་དགོངས་ནས་དེ་སྐད་གསུངས་པ་ཡིན་ན། དེ་ལ་མི་བླུན་པོ་ཁ་ཅིག་སྨན་པའི་རྒྱལ་པོ་འཚོ་བྱེད་ཀྱིས་
དགེ་སློང་མེའུ་དགལ་གྱི་བུ་ལ་ཞེས་མེད་པ་དང་། གུས་པར་བསྒྲིམ་པ་མེད་པ་མ་ཡིན་བར་བསྟན་ཅིང་དྲིས་ན། ཡང་རིངས་པ་རིངས་པ་ཁོ་ནར་སྙུན་གསོས་སྨན་བསྟན་པ་ལས་ན། དེ་བཞིན་གཤེགས་
པ་ནི་སྐུ་ལ་སྙན་མངས་པ་ཡིན་ནོ། །སྙམ་པ་དང་། དེ་བཞིན་གཤེགས་པའི་སྐུ་དེ་ལ་ཡང་ཉེས་དམིགས་དག་སྣང་སྟེ། ཆོས་མཐོང་བ་རྣམས་ལ་ཡང་དེ་ལྟ་བུའི་བག་མེད་པ་དག་འབྱུང་བ་འགྱུར་ན། ར་
