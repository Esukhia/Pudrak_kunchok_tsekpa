༄༅། །ལྡན་འདས་འདི་ལྟ་སྟེ་དཔེར་ན། དཔེར་བགྱི་ན། འཛམ་བུའི་ཆུ་བོའི་གསེར་བརྒྱ་ཚད་པ་མགར་བ་འམ། མགར་བའི་སློབ་མ་མཁས་པས་
ཞི་གའི་ནང་དུ་སྩལ་ཏེ། ཤིན་ཏུ་སྦྱངས་ནས་ལ་བ་སེར་པོའི་ནང་དུ་སྩལ་ན། ཤིན་ཏུ་ཡོངས་སུ་དག་ཅིང་ཡོངས་སུ་བྱང་ལ། སེར་པོར་གདའ་བར་འགྱུར་རོ། །
དེ་བཞིན་དུ་བཅོམ་ལྡན་འདས་ཀྱི་དབང་པོ་རྣམས་ནི་ཤིན་ཏུ་དང་། ཞལ་གྱི་མདོག་ནི་ཡོངས་སུ་དག །པགས་པའི་མདོག་ནི་ཡོངས་སུ་བྱང་ཞིང་སེར་པོར་གདའོ། །བཅོམ་ལྡན་འདས་
བདག་གིས་སྔ་བས་ཀྱང་ཆེས་ལྔ་བ་ན། དེ་བཞིན་གཤེགས་པའི་དབང་པོ་རྣམས་འདི་ལྟར་དང་བ་དང་། ཞལ་གྱི་མདོག་འདི་ལྟར་ཡོངས་སུ་དག་པ་དང་། པགས་
པའི་མདོག་ཡོངས་སུ་བྱང་ཞིང་སེར་བར་གདའ་བར་མ་འཚལ་ཏོ། །བཅོམ་ལྡན་འདས་དེ་ལ་བདག་འདི་སྙམ་དུ་བགྱིད་དེ། དེ་བཞིན་གཤེགས་པ་དི་རིང་འདས་པ་
དང་། མ་བྱོན་པ་དང་། ད་ལྟར་བྱུང་བའི་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་རྗེས་སུ་དགོངས་ཤིང་། དེ་བཞིན་
གཤེགས་པ་དི་རིང་སངས་རྒྱས་ཀྱི་གནས་པས་བཞུགས་ཏེ། རྒྱལ་བའི་གནས་པ་དང་། ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་གནས་པ་དང་། གླང་པོ་ཆེའི་གནས་པས་བཞུགས་
པ་སྙམ་བགྱིད་དོ། །དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀུན་དགའ་བོ་ལེགས་སོ་ལེགས་སོ། །
