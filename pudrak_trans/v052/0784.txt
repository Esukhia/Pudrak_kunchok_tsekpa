དེ་དག་ཟས་ཁམ་གྱིས་ཟས་ཅན་དེ་དྲི་རབ་ཏུ་ཞིམ་པའི་དྲི་དང་། ཁ་དོག་དང་། རོ་དང་། རེག་བྱ་དང་། ཡོན་ཏན་ཕུན་སུམ་ཚོགས་པར་རིག་པར་བྱའོ། །དེ་དག་རེ་རེ་ལ་ཡང་གསེར་གཞོང་གཉིས་
གཉིས་ཡོད་དོ། །དེ་ན་གང་ཟས་གང་འདོད་པ་དེའི་དེ་ཕུན་སུམ་ཚོགས་པར་གཞོང་པ་དེ་ལས་འགྲུབ་པར་འགྱུར་རོ། །ལྷའི་མེ་ཏོག་གི་ཁུ་བ་རོ་དང་ལྡན་པ་མངོན་པར་འགྱུབ་སྟེ། ངེ་ནི་མེ་ཏོག་གི་ཆང་ཞེས་བྱའོ། །དྲང་སྲོང་ཆེན་པོ་ཆང་དེ་ནི་རབ་ཏུ་ཞིམ་པ་མཆོག་ཏུ་བདེ་བ། བསིལ་བ་དྲིས་ཀྱང་ར་རོ་བ་མངོན་པར་འགྱུབ་ན། འཐུངས་པས་ལྟ་ཅི་སྨོས་ཏེ། དེ་ན་གཞོན་ནུར་རོ་བར་འགྱུར་རོ། །དྲང་
སློང་ཆེན་པོ་རྒྱལ་ཆེན་འོ་ཞིའི་རིས་ཀྱིས་དག་གི་གཞལ་མེད་ཁང་ནི་མཛེས་བྱེད་ཅེས་བྱ་སྟེ། རྟག་ཏུ་མེ་ཏོག་རྒྱས་པའི་ཁྲི་ཤིང་གིས་མཛེས་པར་བྱས་པ། གསེར་དང་། དངུལ་དང་། རྡོ་རྡི་འཕྱུར་བའི་ཀུ་
པ་བརྒྱ་སྟོང་གི་ཕྱོགས་ཕྱོགས་ཤིན་ཏུ་བརྟན་པ་ལྷའི་མལ་ཅ་དང་། གཏིང་བ་བཏིང་བས་ཕྱོགས་ཤིན་ཏུ་རྣམ་པར་ཕྱེ་བ། ལྷའི་བུ་མོས་བཟོར་བྱས་པ་སྣ་ཚོགས་ཀྱིས་བརྒྱན་པའི་ཁྱད་བར་དང་ལྡན་པ།
སྟེག་ཚིག་འཛོ་བའི་རྗེས་སུ་ཆགས་པ། ལྷའི་གོས་འབྱུར་ཞིང་རྣམ་པར་མཛེས་པས་སྤྱོད་སྤྱད་པ། ཀ་ལ་བྱིང་ཀ་དང་། བྱུང་བའི་ཚོགས་ཀྱི་སྨྲ་ལྟ་བུ་དང་ལྡན་པ་གཟུགས་རབ་ཏུ་ཆེ་ཞིང་ཤིན་ཏུ་གཞོན་ཤུ་
ཚགས་པ། གླུ་དང་། གར་དང་། བསིལ་སྙན་དང་། འཛམ་ཞིང་རྩེ་བས་རྒྱགས་པའི་ཤེས་ཚེ་ལ་རབ་ཏུ་བགོགས་པ་ཤ་སྟག་གིས་བསྐོར་ཅིང་གང་བ་སྟེ། གནས་དེའི་ཀ་བ་ནི་གསེར་ནུ་དང་།
དངུལ་ཆུས་རྣམ་པར་བྲིས་པ། ངར་གྱི་རྒྱལ་མཚན་སྤྱངས་པ། རྒྱལ་མཚན་དང་། བ་དན་བསྒྲེང་བའོ། །གཞལ་མེད་ཁང་དེ་དག་ན་འདི་ལྟ་སྟེ། ཡུལ་འཁོར་སྲང་དང་། འཕགས་སྐྱེས་པ་དང་།
