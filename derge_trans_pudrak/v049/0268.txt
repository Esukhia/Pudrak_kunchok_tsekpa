རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་དེ་དག་ཐོས་ནས་མངོན་པར་དད། ཡིད་ཆེས། རབ་ཏུ་རྟོག །སོམ་ཉི་མེད། ཐེ་ཚོམ་མི་ཟ། རྒྱ་ཆེར་དགའ་མགུ་སྟེ། རངས་པ་རྙེད་ཅིང་ནམ་མཁའ་ལྟ་བུར་བསམ་གྱིས་མི་ཁྱབ་པའི་ཕྱིར་ངོ་མཚར་དང་རྨད་དུ་འདུ་ཤེས་སྐྱེད་དོ། །དེ་ནས་བཅོམ་ལྡན་འདས་བདེ་བར་གཤེགས་པས་དོན་འདི་བཀའ་སྩལ་ནས། སྟོན་པ་ཡང་འདི་སྐད་གསུང་ངོ་། །འདྲེན་པ་ལ་ནི་འཁྲུལ་པ་ཡོང་མེད་དེ། །ལུས་དང་ངག་རྣམས་དང་ནི་སེམས་མི་གཡོ། །དེ་བཞིན་སྲོག་ཆགས་རྣམས་ལ་ཆོས་སྟོན་ཏེ། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །ཁེངས་པར་མི་འགྱུར་འདུད་པར་མི་འགྱུར་ཏེ། །ཁོང་ཁྲོ་རྗེས་སུ་ཆགས་ལས་མངོན་པར་འདས། །ཉོན་མོངས་མེད་ལ་གནས་ཤིང་ཉོན་མོངས་གྲོལ། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །འདྲེན་པ་ལ་ནི་དྲན་པ་བསྙེལ་བ་མེད། །རྣམ་ཐར་སྤྱོད་ཡུལ་ཆོས་ནི་མངོན་པར་མཁྱེན། །སོ་སོ་ཡང་དག་རིག་པ་བསྙེལ་མེད་དེ། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །མཉམ་པར་གཞག་པས་བཞུགས་ཤིང་བཤོས་ཀྱང་གསོལ། །འཆག་དང་མནལ་བ་ན་ཡང་མཉམ་པར་གཞག །གཡེང་བ་མེད་ཅིང་སེམས་ཅན་འདུ་ཤེས་མེད། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །རྒྱལ་བ་དང་ནི་ཞིང་དང་སེམས་ཅན་ལ། །ཐ་དད་འདུ་ཤེས་བདེ་བར་གཤེགས་ལ་མེད། །བསམ་པ་ཆེན་པོ་མཉམ་པ་ཉིད་ལ་གནས། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །དེ་དག་ལ་ནི་མ་བརྟགས་བཏང་སྙོམས་མེད། །ཤིན་ཏུ་ངེས་པའི་ལམ་ནི་རབ་བསྒོམས་པས། །དེ་ལ་རྟོག་དང་རྣམ་པར་རྟོག་མེད་དེ། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །དགེ་བ་ལ་ནི་འདུན་པ་ཉམས་པ་མེད། །སྟོན་པ་རྟག་ཏུ་སྙིང་རྗེ་རབ་སྦྱོར་ཞིང་། །སེམས་ཅན་རྒྱ་ཆེན་དཔག་ཏུ་མེད་པ་འདུལ། །དེ་ནི་རྒྱལ་བ་རྣམས་ཀྱི་མ་འདྲེས་ཆོས། །ནམ་ཡང་བརྩོན་འགྲུས་ལ་ནི་ཉམས་པ་
