ཤུར་བརྟག་པར་བྱ་སྟེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་དག་ལ་རྟོག་ན་བརྩོན་འགྲུས་རྩོམ་པ་ལ་དགའ་ཞིང་སྐྱོ་བར་ཡང་མི་འགྱུར་རོ། །ཉི་ཤུ་གང་ཞེ་ན། བྱམས་པ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་གཉིད་ཆེ་བ་ནི་རྨུགས་པ་དང་། གཉིད་ཀྱིས་ནོན་ཏེ་ལེ་ལོ་ཆེ་བར་འགྱུར་བ་དང་། ལུས་ཤ་ཆེ་བ་དང་། མདོག་ཡོངས་སུ་མ་དག་པ་དང་། དེའི་འབྱུང་བ་ཆེན་པོ་རྣམས་ནད་བཞིས་ནོན་པར་འགྱུར་བ་དང་། དེའི་མེའི་ཁམས་ཆུང་བར་འགྱུར་བ་དང་། དེའི་ཁ་ཟས་ལེགས་པར་འཇུ་བར་མི་འགྱུར་བ་དང་། དེའི་ལུས་ལ་འབྲས་དང་ཕོལ་མིག་འབྱུང་བར་འགྱུར་བ་དང་། ལེ་ལོ་ཅན་དུ་འགྱུར་བ་དང་། དེའི་གཏི་མུག་གི་དྲ་བ་རྣམ་པར་འཕེལ་བར་འགྱུར་བ་དང་། དེའི་ཤེས་རབ་ཉམ་ཆུང་བར་འགྱུར་བ་དང་། དེ་ཁ་ཟས་འདོད་པ་ལས་ཡིག་འཆུ་བར་འགྱུར་བ་དང་། དེ་གཏི་མུག་གི་མུན་པ་རབ་ཏུ་འཛིན་པར་འགྱུར་བ་དང་། དེ་ལ་མི་མ་ཡིན་པ་རྣམས་བཀུར་སྟི་མི་བྱེད་པ་དང་།སེམས་རྟུལ་བར་འགྱུར་བ་དང་། མི་ལྡང་བ་མང་བར་འགྱུར་བ་དང་། དེའི་སེམས་བག་ལ་ཉལ་གྱིས་ཀུན་ནས་དཀྲིས་ཤིང་འདུག་པར་འགྱུར་བ་དང་། དེ་དགེ་བའི་ཆོས་ལ་འདུན་པར་མི་འགྱུར་བ་དང་། དགེ་བའི་ཆོས་ཐམས་ཅད་ལས་ཉམས་པར་འགྱུར་བ་དང་། སྤྱོད་པ་ཞུམ་པར་གྱུར་ནས་བརྩོན་འགྲུས་རྩོམ་པ་རྣམས་ཀྱི་སྤྱོད་པ་ལ་སྨོད་པ་དང་། འཁོར་གྱི་ནང་དུ་སོང་ན་བརྙས་པར་འགྱུར་བ་སྟེ། བྱམས་པ་ཉི་ཤུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་གཉིད་ལ་དགའ་ཞིང་རྨུགས་པ་དང་། གཉིད་མང་བའི་ཉེས་དམིགས་ཡིན་ཏེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་དག་ལ་རྟོག་ན་བརྩོན་འགྲུས་རྩོམ་པ་ལ་དགའ་ཞིང་ཡོངས་སུ་སྐྱོ་བར་ཡང་མི་འགྱུར་རོ། །དེ་ནས་དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ཚིགས་སུ་བཅད་པ་འདི་དག་བཀའ་སྩལ་ཏོ། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ཡི་ལུས་ནི་ཤ་ཆེ་མཁྲེགས་མ་ཡིན། །ལེ་ལོ་དེ་ཆེ་བླུན་ཞིང་འཐུན་མི་བྱེད། །དེ་ཡི་མདོག་ཀྱང་དག་པར་མི་འགྱུར་རོ། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །བད་ཀན་རླུང་དང་དེ་བཞིན་མཁྲིས་པ་རྣམས། །དེ་དག་དེ་ཡི་ལུས་ལ་ཤས་ཆེར་འབྱུང་། །དེ་ཡི་ཁམས་རྣམས་རབ་ཏུ་འཁྲུག་པར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་
འགྱུར་བ། །ཁ་ཟས་ཉེས་པས་ལྟོ་ནི་མ་དག་ཅིང་། །ལུས་ཀྱང་ལྕི་ལ་ཟོ་བདོག་བདེ་མ་ཡིན། །དེ་ཡི་ཚིག་ཀྱང་གསལ་བར་མི་འགྱུར་རོ། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ཡི་ལུས་ལ་ཕོལ་མིག་མང་པོ་འབྱུང་། །དེ་ནི་ཉིན་དང་མཚན་དུ་བདེ་བར་ཉལ། །དེ་ལ་རྨ་རྣམས་མི་བཟད་འབྱུང་བར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །ལེ་ལོ་བྱེད་ཅིང་བརྩོན་འགྲུས་ཡོངས་སུ་ཉམས། །དེ་ནི་བདེ་ལས་ཉམས་ཤིང་ནོར་རྣམས་འགྲིབས། །དེ་ནི་རྟག་ཏུ་ཞན་ཅིང་སེམས་ཀྱང་རྨོངས། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །གཏི་མུག་དྲ་བ་ཆེན་པོ་འཕེལ་འགྱུར་ཞིང་། །དེ་ནི་ཐེ་ཚོམ་ཅན་དེ་ལྟ་བ་ཐོབ། །དེ་ནི་ལྟ་བར་གྱུར་པ་མང་བར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ཡི་ཤེས་རབ་ཤིན་ཏུ་ཉམ་ཆུང་འགྱུར། །དེ་ཡི་བློ་ཡང་ཡོངས་སུ་ཉམས་པར་འགྱུར། །ཡེ་ཤེས་ལས་ནི་རྟག་ཏུ་དེ་ཉམས་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །མི་ཤེས་ལེ་ལོ་སྙོམ་ལས་ཤེས་རབ་འཆལ། །མི་མ་ཡིན་པ་དག་གིས་གླགས་རྙེད་དེ། །ནགས་ན་གནས་པའི་ཚེ་ནི་འཚེ་བར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །སེམས་ཀྱང་རྟུལ་ཞིང་དེ་ཡི་དྲན་པ་རྨོངས། །ལུང་ནོད་ཁ་ཏོན་བྱེད་ལ་དེ་མི་གནས། །ཆོས་སྨྲ་ན་ཡང་དེ་ནི་རྟག་ཏུ་ཐོགས། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ཡི་ཉོན་མོངས་ཀུན་ནས་ལྡང་བར་འགྱུར། །རྣམ་པར་འཁོར་ཞིང་གཡོ་ལ་རྨོངས་པ་དང་། །དེ་ཡི་སེམས་ནི་རྟག་ཏུ་གདུང་བར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ཡི་བྱ་བ་མང་པོ་ཉམས་ཤིང་འགགས། །ལུས་ཀྱང་གདུངས་ཤིང་ཡིད་ཀྱང་ཡོངས་སུ་གདུངས། །བག་ལ་ཉལ་བ་འཕེལ་བར་ངེས་མི་ཤེས། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ནི་དགེ་སེམས་རྟག་ཏུ་མི་འདོད་དེ། །དེ་ནི་ཆོས་ལ་འདུད་པ་སྐྱེད་མི་བྱེད། །ཕྱིར་ཞིང་ཆོས་མ་ཡིན་ལ་དེ་དགར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །དེ་ནི་རྨོངས་ཤིང་ཆོས་ལ་འདུན་པ་ཉམས། །བྱིས་པ་ཡོན་ཏན་ཀུན་ལས་ཡོངས་སུ་ཉམས། །དཀར་པོ་འཇོམས་ཤིང་མུན་པར་འགྲོ་བར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །འཇིགས་པར་འགྱུར་ཞིང་སེམས་ཀྱང་རབ་ཏུ་ཞུམ། །དེ་ནི་རྟག་པར་མཆོག་ཏུ་དགའ་མི་འགྱུར། །གཉིད་ཀྱིས་ནོན་པ་ལུས་ཀྱང་ལྷོད་པར་འགྱུར། །གང་ཞིག་རྨུགས་དང་གཉིད་ལ་དགའ་འགྱུར་བ། །བདག་ཉིད་མི་ཤེས་ལེ་ལོ་ཅན་གྱུར་ནས། །བརྩོན་འགྲུས་སྟོབས་དང་ལྡན་ལ་ཕྲག་དོག་བྱེད། །བརྩོན་འགྲུས་ལྡན་པ་རྣམས་ལ་མི་སྙན་སྨྲ། །ཉེས་པ་འདི་འདྲ་འདི་དག་རྟོགས་གྱུར་ནས། །མཁས་པ་སུ་ཞིག་རྨུགས་གཉིད་དགའ་བར་བྱེད། །ཤིན་ཏུ་གཏི་མུག་ལྟ་བ་མང་པོའི་རྒྱུ། །དོན་བཟང་དམ་ཆོས་ཡོན་ཏན་ཆུད་གཟོན་པ། །གཉིད་འདི་བརྩོན་འགྲུས་འཕྲོག་པར་རྟོགས་གྱུར་ནས། །མཁས་པ་ཡོན་ཏན་ལྡན་པ་སུ་དེར་དགའ། །སྡུག་བསྔལ་ཀུན་དང་མུན་པ་སེལ་བྱེད་ཅིང་། །ངན་སོང་ཡོངས་སུ་སྤོང་བའི་གཞིར་གྱུར་པ། །སངས་རྒྱས་ཀུན་གྱིས་བསྔགས་པའི་བརྩོན་འགྲུས་
འཕགས། །གང་ཡིན་དེ་ནི་རྟག་ཏུ་བརྟེན་པར་གྱིས། །འཇིག་རྟེན་འདི་ན་བཟོ་གང་ཅི་ཡོད་དང་། །འཇིག་རྟེན་འདས་བཟོ་གང་དག་ཅི་ཡོད་པ། །བརྩོན་འགྲུས་བརྩམས་ལ་རྙེད་པར་མི་དཀའ་ན། །མཁས་པ་སུ་དག་བརྩོན་འགྲུས་སྟོབས་ཀྱིས་སྐྱོ། །གང་དག་སངས་རྒྱས་བྱང་ཆུབ་ཞུགས་གྱུར་པ། །དེ་དག་རྨུགས་གཉིད་སྐྱོན་རྣམས་མཐོང་ནས་ནི། །རྟག་ཏུ་བརྩོན་འགྲུས་རྩོམ་ལ་གནས་པར་བྱེད། །འདི་ནི་ང་ཡིས་དེ་དག་བསྐུལ་བྱས་ཡིན། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་གང་དག་རྨུགས་པ་དང་གཉིད་ཀྱི་སྐྱོན་འདི་ལྟ་བུ་འདི་དག་ཐོས་ནས་སྤོང་བར་མི་བགྱིད་ཅིང་སྐྱོ་བར་མི་བགྱིད་པ་དེ་དག་དང་བརྩོན་འགྲུས་རྩོམ་པ་མི་སྐྱེད་པ་དེ་དག་དེ་ཙམ་དུ་རྨོངས་པར་འགྱུར་བ་ནི་ངོ་མཚར་ལགས་སོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་སློབ་པར་འཚལ་བ་དང་། ལྷག་པའི་བསམ་པས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འཚལ་བ་སུ་ཞིག་ཡང་དག་པའི་གནས་འདི་ལྟ་བུ་འདི་དག་ཐོས་ན་ལེ་ལོ་བགྱིད་ཅིང་དགེ་བའི་ཆོས་རྣམས་སྐྱེད་པར་མི་བགྱིད། བརྩོན་འགྲུས་ཀྱི་ཡོན་ཏན་དང་། ཕན་ཡོན་འདི་ལྟ་བུ་ཐོས་ནས་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ཡོངས་སུ་རྫོགས་པར་བགྱི་བའི་སླད་དུ་སུ་ཞིག་བརྩོན་འགྲུས་རྩོམ་པར་མི་བགྱིད་དེ། །དེ་བཞིན་གཤེགས་པས་གཉིད་ལ་དགའ་བའི་ནོངས་དམིགས་དང་། བརྩོན་འགྲུས་རྩོམ་པའི་ཕན་ཡོན་འདི་དག་ལེགས་པར་བཀའ་སྩལ་ལགས་སོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔས་ལས་ལ་དགའ་ཞིང་འཇིག་རྟེན་པའི་སྦག་པས་སྦགས་པའི་ནོངས་དམིགས་ཇི་ལྟར་བརྟགས་པར་བགྱི། བྱང་ཆུབ་སེམས་དཔའ་གང་དག་ལ་བརྟགས་ན་འདོད་པ་ཆུང་ཞིང་བགྱི་བ་ཉུང་ལ། དམ་པའི་ཆོས་བཙལ་བའི་སླད་དུ་རྟག་ཏུ་བརྩོན་ཞིང་སྦྱོར་བ་ལ་གནས་པར་འགྱུར་བ་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། བྱམས་པ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོས་ལས་ལ་དགའ་བའི་ཉེས་དམིགས་ཉི་ཤུར་བརྟག་པར་བྱ་སྟེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་དག་ལ་རྟོག་ན་འདོད་པ་ཆུང་ཞིང་བྱ་བ་ཉུང་ལ་དམ་པའི་ཆོས་བཙལ་བའི་ཕྱིར་རྟག་ཏུ་བརྩོན་ཞིང་སྦྱོར་བ་ལ་གནས་པར་འགྱུར་རོ། །ཉི་ཤུ་གང་ཞེ་ན། བྱམས་པ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ལས་ལ་དགའ་བ་འཇིག་རྟེན་པའི་ལས་རྣམས་ལ་བརྩོན་པར་འགྱུར་བ་དང་།
 ལས་ངན་པ་ཐམས་ཅད་ལ་གནས་པར་འགྱུར་བ་དང་། ལུང་ནོད་པ་དང་། ཁ་དོན་ལ་བརྩོན་པ་དག་གིས་བརྙས་པར་འགྱུར་བ་དང་། ནང་དུ་ཡང་དག་འཇོག་ལ་དགའ་བ་རྣམས་ཀྱིས་སྨད་པར་འགྱུར་བ་དང་། འཁོར་བ་ཐོག་མ་དང་ཐ་མ་མེད་པ་ཡོངས་སུ་འཛིན་པའི་ལས་རྩོམ་པར་འགྱུར་བ་དང་། བྲམ་ཟེ་དང་ཁྱིམ་བདག་དད་པ་རྣམས་ཀྱི་ཡོན་སྦྱོང་བར་མི་བྱེད་པ་དང་། དེའི་སེམས་དངོས་པོར་འཛིན་པ་ལ་འཇུག་པར་འགྱུར་བ་དང་། ཆགས་པ་མང་བར་འགྱུར་བ་དང་། ཁྱིམ་སོ་རྣམས་ལ་རྒྱུན་དུ་སྦྱོར་བར་འགྱུར་བ་དང་། དབང་བྱེད་པར་མི་འགྱུར་བ་དང་། ཟློག་བྱེད་ཀྱི་ངང་ཚུལ་ཅན་ཡིད་སྨོན་མང་པོ་ལ་རྒྱུན་དུ་གནས་པ་དང་། ཁྱིམ་གྱི་བྱ་བ་དག་ལ་བརྩོན་པ་དང་། དེ་རོ་དག་ལ་ཤས་ཆེར་འདུན་ཅིང་ཆགས་པ་སྐྱེད་པ་དང་། ལེན་པ་མེད་པར་མི་དགའ་བ་དང་། ཀུན་ནས་མནར་སེམས་མང་བའི་ལས་ཀྱི་སྒྲིབ་པ་རྒྱུན་དུ་མངོན་པར་འདུ་བྱེད་པ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མ་ལ་རྟོན་པར་འགྱུར་བ་དང་། དེ་ཟས་ཀྱི་རྣམ་པར་རྟོག་པས་ཉིན་མཚན་འདའ་བར་བྱེད་པ་དང་། འཇིག་རྟེན་པའི་བྱ་བ་དག་རྒྱུན་དུ་འདྲི་བར་བྱེད་པ་དང་། མི་རིགས་པའི་སྨྲ་བ་དག་ལ་མངོན་པར་དགའ་བར་འགྱུར་བ་དང་། ཞལ་ཏ་བྱེད་པའི་རྒྱགས་པས་ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྣམས་དབང་ཕྱུག་གིས་ཚར་གཅོད་པར་བྱེད་པ་དང་། ཕ་རོལ་གྱི་འཁྲུལ་པ་ལ་རྟོག་ཅིང་བདག་གི་འཁྲུལ་པ་ལ་མི་རྟོག་པར་འགྱུར་བ་དང་། དོན་དང་ལྡན་ཞིང་རྣམ་པར་ངེས་པའི་གཏམ་དག་ལ་སྐྱེན་པར་བརྙས་པར་འགྱུར་བ་སྟེ། བྱམས་པ་ཉི་ཤུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་ལས་ལ་དགའ་ཞིང་ཞལ་ཏ་བྱེད་པའི་སྦག་པས་སྦགས་པའི་ཉེས་དམིགས་ཡིན་ཏེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་དག་ལ་རྟོག་ན་འདོད་པ་ཆུང་ཞིང་བྱ་བ་ཉུང་བར་འགྱུར་ལ་འཇིག་རྟེན་པའི་བྱ་བ་སྤངས་ནས་འཇིག་རྟེན་ལས་འདས་པའི་བྱ་བ་ལ་བརྩོན་པར་འགྱུར་རོ། །དེ་ནས་དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ཚིགས་སུ་བཅད་པ་འདི་དག་བཀའ་སྩལ་ཏོ། །དེ་ནི་ལས་ངན་དག་ལ་གནས་འགྱུར་ཞིང་། །ལས་མཆོག་དག་ལས་དེ་ནི་རྒྱང་ཡང་རིང་། །འདིར་ནི་རྙེད་པ་ཆེ་ལས་རིང་བར་འགྱུར། །རྣམ་པར་ཆགས་སེམས་ཉེས་པ་འདི་དག་ཡིན། །དགེ་སློང་ལུང་ནོད་ཁ་ཏོན་དགའ་རྣམས་ཀྱིས། །དེ་ལ་བརྙས་ཤིང་འདོད་པར་མི་བྱེད་ལ། །བསམ་གཏན་བྱེད་པ་རྣམས་ཀྱིས་སྨད་པར་འགྱུར། །རྟག་ཏུ་ལས་གནས་སྐྱོན་ནི་འདི་དག་ཡིན། །འཁོར་བའི་ལས་ལ་མངོན་པར་བརྩོན་པ་དང་། །ཐར་པ་ལས་ནི་རྒྱང་རིང་བཅིངས་པར་གནས། །དེ་ནི་ཡུལ་འཁོར་ཟས་ལ་མི་རིགས་སྤྱོད། །ཉེས་པ་འདི་དག་ལས་ལ་ཆགས་པའི་ཡིན། །རྟག་ཏུ་དངོས་པོ་དག་ནི་ལེན་པར་བྱེད། །ཡོངས་སུ་འཛིན་པ་མེད་ལ་དགའ་མི་སྐྱེད། །ནོར་མེད་ནོར་ནི་དོན་དུ་གཉེར་བར་བྱེད། །ཉེས་པ་འདི་དག་ལས་ལ་ཆགས་པའི་ཡིན། །ཁྱིམ་དུ་རྒྱུ་བ་དག་ལ་ཆགས་པར་འགྱུར། །འདུ་འཛི་དགའ་བས་ཐམས་ཅད་འཇིག་པར་བྱེད། །གཟེབ་ཀྱི་ནང་དུ་བྱ་བཅུག་བཞིན་དུ་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་ཆགས་པའི་ཡིན། །ཁྱིམ་པའི་ལས་ལ་དེ་རྟག་མྱ་ངན་བྱེད། །ཡོངས་སུ་གདུངས་པར་གྱུར་ཀྱང་ཡིད་མི་འབྱུང་། །ནམས་ཀྱང་ཚིག་ནི་བཙུན་པར་མི་འགྱུར་ཏེ། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །བླ་མས་བསྐུལ་ན་བཀའ་བློ་མི་བདེ་ཞིང་། །རྗེས་སུ་བསྟན་པ་འཐུན་པར་འཛིན་མི་བྱེད། །དེ་ནི་མྱུར་དུ་ཚུལ་ཁྲིམས་ཉམས་པར་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །ཁྱིམ་ན་གནས་པའི་ལས་རྣམས་རྟག་སེམས་པས། །དེ་ནི་རྟག་ཏུ་འདོད་པར་འགྱུར་བ་སྟེ། །བསམ་གཏན་དང་ནི་སྤོང་བ་དེ་མི་བྱེད། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་ཡི་འདོད་ཆགས་ཤས་ནི་ཆེ་བར་སྐྱེ། །དེ་ནི་རོ་བྲོ་རྣམས་ལ་ཆགས་ཤིང་ཞེན། །དེ་ནི་ངན་ངོན་ཙམ་གྱིས་དགར་མི་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་ནི་འཁོར་ཕལ་ཆེ་ན་དགར་འགྱུར་ཏེ། །དེ་ཉམས་ན་ནི་ཤིན་ཏུ་སྡུག་བསྔལ་འགྱུར། །དེ་ནི་བོང་བུ་བཞིན་དུ་ལང་ལོང་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་སེམས་རྟག་ཏུ་གནོད་སེམས་ལྡན་པར་འགྱུར། །ལས་རྣམས་འཕེལ་ཞིང་ཟད་པར་མི་འགྱུར་ཏེ། །སྲེད་པའི་འཁྲི་ཤིང་དག་གིས་དམ་དུ་བཅིངས། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་ནི་རྟག་ཏུ་བླ་མ་རྣམས་མི་སྟེན། །ཕྱོགས་འདོད་ཁྱིམ་པ་ལ་ནི་སྟེན་པར་བྱེད། །ཚུལ་ཁྲིམས་གནས་པ་དག་ལའང་ཕྲག་དོག་བྱེད། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །ཉིན་དང་མཚན་དུ་སེམས་གཞན་མེད་པར་ནི། །ཟས་དང་
གོས་ལ་རྒྱུན་དུ་སེམས་པར་བྱེད། །དེ་ནི་རྟག་ཏུ་ཡོན་ཏན་འདོད་མ་ཡིན། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་ནི་འཇིག་རྟེན་པ་ཡི་བྱ་བ་འདྲི། །རིགས་པ་མིན་པའི་སྨྲ་བ་རྣམས་ལ་དགའ། །རིགས་པའི་སྨྲ་ལ་དེ་ནི་དགའ་མི་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །སྡོམ་བརྩོན་རྣམས་ལ་དེ་ནི་རྟག་ཏུ་གཤེ། །དབང་ཕྱུག་ཉེས་པས་དེ་དག་དེས་ཚར་གཅོད། །མི་སྡུག་ལས་རྣམས་མངོན་པར་འདུ་བྱེད་དེ། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །དེ་ནི་རྟག་ཏུ་གཞན་གྱི་སྐྱོན་རྟོག་ཅིང་། །བདག་ལ་ཡོད་པའི་སྐྱོན་རྣམས་མི་མཐོང་ལ། །ཡོན་ཏན་ལྡན་པ་རྣམས་ལ་འཁྲུལ་པ་ཚོལ། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །བྱིས་པ་དེ་ནི་གཞན་གྱིས་མྱུར་དུ་བརྙས། །དེ་ནི་ཕ་རོལ་དག་ལ་ཆོས་སྟོན་ཚེ། །བློ་ཉམས་པས་ན་མངོན་པར་ཤེས་མི་འགྱུར། །ཉེས་པ་འདི་དག་ལས་ལ་དགའ་བའི་ཡིན། །གནོད་འགྱུར་ཉེས་པ་འདི་དག་མཐོང་ནས་ནི། །ལས་ངན་དག་ལ་སུ་ཞིག་དགའ་བར་བྱེད། །དཔའ་བོ་དག་ནི་རྟག་ཏུ་ལས་མཆོག་བྱེད། །བྱས་པ་གང་ཡིན་ཆུད་ནི་གཟོན་མི་བྱེད། །ལན་སྟོང་བརྗོད་ཀྱང་ཚིག་གཅིག་མི་འཛིན་པ། །དེ་ནི་བློ་ཅན་དག་གིས་སྤྱོ་བ་ཡིན། །གང་ཞིག་ལས་ངན་དག་ནི་བྱེད་པ་ལ། །དམ་པ་དག་གིས་དེ་ལྟར་རྟག་ཏུ་སྨད། །དེ་ལྟ་བས་ན་མཁས་མིས་འདི་རྟག་ཏུ། །ལས་ངན་དག་ནི་རྣམ་པར་སྤངས་ནས་ཀྱང་། །ཐུབ་དབང་རྣམས་ཀྱིས་རྟག་ཏུ་བསྔགས་པ་ཡི། །བཟང་པོའི་ལས་རྣམས་གང་ཡིན་དེ་དག་བྱ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་གང་དག་ཆོས་ཀྱི་མཆོག་ཡོངས་སུ་སྤངས་ནས་ལས་ངན་པ་རྩོམ་པ་དེ་དག་ནི་ཤེས་རབ་ཤིན་ཏུ་ཆུང་ཞིང་ཤེས་རབ་ཀྱིས་དམན་པར་འགྱུར་བ་ལགས་སོ། །དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །བྱམས་པ་དེ་དེ་བཞིན་ཏེ། དེ་དེ་བཞིན་ནོ། །ཇི་སྐད་སྨྲས་པ་བཞིན་ཏེ། བྱང་ཆུབ་སེམས་དཔའ་གང་དག་ཆོས་ཀྱི་མཆོག་ཡོངས་སུ་སྤངས་ནས་ལས་ངན་པ་རྩོམ་པ་དེ་དག་ནི་ཤེས་རབ་ཤིན་ཏུ་ཆུང་ཞིང་ཤེས་རབ་ཀྱིས་དམན་པར་འགྱུར་བ་ཡིན་ནོ། །བྱམས་པ་གཞན་ཡང་ཁྱོད་མོས་པར་བྱའོ། །ཁྱོད་ཀྱིས་ཁོང་དུ་ཆུད་པར་བྱའོ། །བྱང་ཆུབ་སེམས་དཔའ་གང་དག་ལ་སྦྱོར་བ་མེད་ཅིང་བསམ་གཏན་མེད་ལ་སྤོང་བ་མེད་ཅིང་འདོན་པ་མེད་པ་དང་། མང་དུ་ཐོས་པ་ཡོངས་སུ་ཚོལ་བ་མེད་པ་དེ་དག་ནི་དེ་བཞིན་གཤེགས་པའི་བསྟན་པ་ལ་རབ་ཏུ་བྱུང་བ་མ་ཡིན་ནོ། །བྱམས་པ་དེ་བཞིན་གཤེགས་པའི་བསྟན་པ་ནི་བསམ་གཏན་དང་སྤོང་བས་ཕྱེ་བ་དང་། ཡེ་ཤེས་ཀྱིས་འདུས་བྱས་པ་དང་། ཡེ་ཤེས་ཀྱིས་མཉམ་པར་བཞག་པ་དང་། བརྩོན་པས་ཕྱེ་བ་ཡིན་གྱི། ཁྱིམ་པའི་ལས་དང་ཞལ་ཏ་བྱེད་པས་ཕྱེ་བ་ནི་མ་ཡིན་ཏེ། འདི་ལྟར་ཞལ་ཏ་བྱེད་པའི་ལས་དེ་ནི་མི་རིགས་པ་ལ་སྦྱོར་བ་རྣམས་དང་། འདུ་བྱེད་ལ་མངོན་པར་དགའ་བ་རྣམས་དང་། འཇིག་རྟེན་པའི་བྱ་བས་སྦགས་པ་སྟེ། བྱང་ཆུབ་སེམས་དཔས་དེ་ལ་འདོད་པ་བསྐྱེད་པར་མི་བྱའོ། །བྱམས་པ་གལ་ཏེ་བྱང་ཆུབ་སེམས་དཔའ་ཞལ་ཏ་བྱེད་པ་ལ་བརྩོན་པས་རིན་པོ་ཆེ་སྣ་བདུན་གྱི་མཆོད་རྟེན་དག་གིས་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་གྱི་ཁམས་འདི་བཀང་ན་དེས་ནི་ང་མཉེས་པར་བྱས་པ་མ་ཡིན། རི་མོར་བྱས་པ་མ་ཡིན། བཀུར་སྟིར་བྱས་པ་མ་ཡིན་གྱི། བྱམས་པ་གལ་ཏེ་བྱང་
ཆུབ་སེམས་དཔའ་ཐ་ན་ཕ་རོལ་ཏུ་ཕྱིན་པ་དང་ལྡན་པའི་ཚིག་བཞི་པའི་ཚིགས་སུ་བཅད་པ་གཅིག་ཙམ་འཛིན་ཏམ། འཆང་ངམ། ཀློག་གམ། ཀུན་ཆུབ་པར་བྱེད་དམ། འཆང་ན་དེས་ནི་ང་ལ་བཀུར་སྟིར་བྱས། བླ་མར་བྱས། རི་མོར་བྱས། མཆོད་པ་བྱས་པར་འགྱུར་རོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། བྱམས་པ་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་བྱང་ཆུབ་ནི་མང་དུ་ཐོས་པ་ལས་བྱུང་བ་ཡིན་གྱི། དངོས་པོ་ཡོངས་སུ་འཛིན་པ་ལས་བྱུང་བ་ནི་མ་ཡིན་པའི་ཕྱིར་རོ། །བྱམས་པ་གལ་ཏེ་བྱང་ཆུབ་སེམས་དཔའ་ཞལ་ཏ་བྱེད་པ་ལ་བརྩོན་པས་བྱང་ཆུབ་སེམས་དཔའ་ལུང་ནོད་པ་དང་། ཁ་ཏོན་བྱེད་པ་ལ་བརྩོན་པ་ཞལ་ཏ་བྱེད་པའི་སྦག་པས་སྦགས་པར་བྱེད་ན་བསོད་ནམས་འདོད་པ་དེས་ནི་བསོད་ནམས་མ་ཡིན་པའི་ཕུང་པོ་ཆེན་པོ་སྐྱེད་པར་འགྱུར་ཏེ། ལས་ཀྱི་སྒྲིབ་པ་ཡང་ཡོངས་སུ་འཛིན་པར་འགྱུར་རོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། བསོད་ནམས་བྱ་བའི་གཞི་གསུམ་པོ་གང་ཡིན་པ་དེ་དག་ཐམས་ཅད་ནི་ཤེས་རབ་ལས་བྱུང་བའི་ཕྱིར་རོ། །བྱམས་པ་དེ་ལྟ་བས་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞལ་ཏ་བྱེད་པ་ལ་བརྩོན་པས་བྱང་ཆུབ་སེམས་དཔའ་ལུང་ནོད་པ་དང་། ཁ་ཏོན་བྱེད་པ་ལ་བརྩོན་པའི་བར་ཆད་མི་བྱའོ། །འཛམ་བུའི་གླིང་འདི་ཞལ་ཏ་བྱེད་པས་ཡོངས་སུ་གང་བར་གྱུར་པའི་ཞལ་ཏ་བྱེད་པ་དེ་དག་ཐམས་ཅད་ཀྱིས་ནི་བྱང་ཆུབ་སེམས་དཔའ་ལུང་ནོད་པ་དང་། ཁ་ཏོན་བྱེད་པ་ལ་བརྩོན་པ་གཅིག་ལ་རིམ་གྲོ་དང་བསྙེན་བཀུར་བྱའོ། །བྱང་ཆུབ་སེམས་དཔའ་ལུང་ནོད་པ་དང་། ཁ་ཏོན་བྱེད་པ་ལ་བརྩོན་པ་འཛམ་བུའི་གླིང་གང་བསྙེད་ཀྱིས་ནི་བྱང་ཆུབ་སེམས་དཔའ་ནང་དུ་ཡང་དག་འཇོག་ལ་བརྩོན་པ་གཅིག་ལ་རིམ་གྲོ་དང་བསྙེན་བཀུར་བྱའོ། །དེ་ལྟར་བྱས་ན་དེ་བཞིན་གཤེགས་པས་རྗེས་སུ་ཡི་རང་བ་དང་གནང་བ་ཡིན་ཏེ། དེ་དག་གི་བསྙེན་བཀུར་ནི་ལེགས་པར་བྱས་པ་ཡིན་ནོ། །གང་ཞིག་ཤེས་རབ་སྤྱོད་པ་ལ་མོས་ནས་རིམ་གྲོ་དང་བསྙེན་བཀུར་བྱེད་པའི་བསོད་ནམས་ཀྱི་ཕུང་པོ་ནི་ཚད་མེད་པར་སྐྱེད་པར་འགྱུར་རོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན།འདི་ལྟར་ཤེས་རབ་ཀྱི་ལས་ནི་དཀའ་བའི་ལས་དང་། བླ་ན་མེད་པ་དང་། གོང་ན་མེད་པ་དང་། འཇིག་རྟེན་གསུམ་པོ་ཐམས་ཅད་ལས་ཁྱད་པར་དུ་འཕགས་ཤིང་མངོན་པར་འཕགས་པའི་ཕྱིར་རོ། །བྱམས་པ་དེ་ལྟ་བས་ན་བྱང་ཆུབ་སེམས་དཔའ་མངོན་པར་བརྩོན་པ་དོན་དུ་གཉེར་ཞིང་བརྩོན་འགྲུས་རྩོམ་པར་འདོད་པས་ཤེས་རབ་ལ་བརྩོན་པར་བྱ་ཞིང་ཤེས་རབ་ཡོངས་སུ་བཙལ་བ་ལ་བརྩོན་འགྲུས་བརྩམ་པར་བྱའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་འདུ་འཛི་ལ་དགའ་བ་དང་།
 ལས་ལ་དགའ་བ་རྣམས་ཀྱི་ནོངས་དམིགས་རྣམས་ནི་བཀའ་སྩལ་ལགས་ན། བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔས་སྤྲོས་པ་ལ་དགའ་བའི་ནོངས་དམིགས་རྣམས་ཇི་ལྟར་བརྟག་པར་བགྱི།བྱང་ཆུབ་སེམས་དཔས་གང་དག་ལ་བརྟགས་ན་ཉེ་བར་ཞི་བ་ལ་གནས་ཤིང་རྩོད་པ་མ་མཆིས་པར་གནས་པར་འགྱུར་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། བྱམས་པ་བྱང་ཆུབ་སེམས་དཔས་སྤྲོས་པ་ལ་དགའ་བའི་ཉེས་དམིགས་རྣམས་མདོར་བསྡུ་ན་ཉི་ཤུ་པོ་འདི་དག་ཏུ་བརྟག་པར་བྱའོ། །རྒྱས་པར་ནི་དེ་བཞིན་གཤེགས་པས་མཐའ་ཡས་པར་བསྟན་ཏེ་མཐར་ཐུག་པར་ཡང་མི་འགྱུར་རོ། །ཉི་ཤུ་གང་ཞེ་ན། བྱམས་པ་བྱང་ཆུབ་སེམས་དཔའ་སྤྲོས་པ་ལ་དགའ་བ་ནི་ཚེ་འདི་ཉིད་ལ་སྡུག་བསྔལ་བ་དང་ཡིད་མི་བདེ་བ་མང་བར་གནས་པ་དང་། བཟོད་པ་ལས་ཡོངས་སུ་ཉམས་པར་འགྱུར་བ་དང་། དེའི་མི་མཛའ་བ་རྣམས་ཡི་རངས་པར་འགྱུར་བ་དང་། དེ་ལ་བདུད་ཀྱང་མགུ་བར་འགྱུར་བ་དང་། དེ་ལ་བདུད་ཀྱི་འཁོར་ཀྱང་དགའ་བར་འགྱུར་བ་དང་། དེའི་དགེ་བའི་ཆོས་མ་སྐྱེས་པ་རྣམས་སྐྱེ་བར་མི་འགྱུར་བ་དང་། སྐྱེས་པ་རྣམས་མེད་པར་འགྱུར་བ་དང་། འཐབ་པ་དང་འཁོན་དུ་འཛིན་པ་དང་། རྩོད་པ་དང་། འགྱེད་པ་མང་བར་འགྱུར་བ་དང་། འཐབ་པ་དང་། རྩོད་པ་དང་། འགྱེད་པས་ངན་སོང་དུ་འགྲོ་བར་འགྱུར་བའི་ལས་མངོན་པར་འདུ་བྱེད་པར་འགྱུར་བ་དང་། བག་མེད་པར་འགྱུར་བའི་ལམ་རབ་ཏུ་འཛིན་པ་དང་། དེའི་ལྕེའི་དབང་པོ་རྩུབ་པར་འགྱུར་བ་དང་། དེའི་ལུང་མནོས་པའི་ཆོས་རྣམས་ལེགས་པར་གནས་པར་མི་འགྱུར་བ་དང་། མདོ་སྡེ་ལུང་མ་མནོས་པ་རྣམས་རྣ་ལམ་དུ་གྲག་པར་མི་འགྱུར་བ་དང་། དགེ་བའི་བཤེས་གཉེན་ཐམས་ཅད་ཀྱིས་ཡོངས་སུ་སྤོང་བར་འགྱུར་བ་དང་། སྡིག་པའི་གྲོགས་པོ་རྣམས་དང་མྱུར་དུ་ཕྲད་པར་འགྱུར་བ་དང་། དཀའ་བའི་ལམ་གྱིས་ངེས་པར་འབྱུང་བར་འགྱུར་བ་དང་། རྒྱུན་མི་འཆད་པར་ཡིད་དུ་མི་འོང་བའི་ཚིག་ཐོས་པར་འགྱུར་བ་དང་། གང་དང་གང་དུ་སྐྱེས་པ་དེ་དང་དེར་ཐེ་ཚོམ་ཅན་དུ་འགྱུར་བ་དང་། དེ་ལ་མི་ཁོམ་པ་བརྒྱད་པོ་དག་ཉེ་བར་འགྱུར་བ་དང་། རྟག་ཏུ་དེའི་དཀར་པོའི་ཕྱོགས་ལ་བརྩོན་པ་ལ་བར་ཆད་འབྱུང་བར་འགྱུར་བ་སྟེ། བྱམས་པ་ཉི་ཤུ་པོ་དེ་དག་ནི་སྤྲོས་པ་ལ་དགའ་བའི་ཉེས་དམིགས་རྣམས་མདོར་བསྡུས་པ་ཡིན་ཏེ།དེ་དག་ཡོངས་སུ་སྤང་བར་བྱའོ། །དེ་ནས་དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ཚིགས་སུ་བཅད་པ་འདི་དག་བཀའ་སྩལ་ཏོ། །དེ་ནི་ཚེ་འདིར་བདེ་བར་མི་འགྱུར་ཏེ། །བཟོད་དང་རིང་ཞིང་ཁྲོ་བའི་ཕྱོགས་ལ་གནས། །དེ་ཡི་དགྲ་རྣམས་དགའ་བར་འགྱུར་བ་སྟེ། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །བདུད་ཀྱང་དེ་ལ་དགའ་ཞིང་མགུ་བར་འགྱུར། །བདུད་ཀྱི་ཕྱོགས་ཀྱང་དེ་ལ་དགའ་བར་འགྱུར། །དེ་ཡི་དགེ་བའི་ཆོས་སྐྱེས་ཆུད་ཟར་འགྱུར། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །དེ་ནི་ནམས་ཀྱང་དགེ་བ་མི་སྐྱེད་དེ། །འཆི་ཀར་ཐ་དད་རྒྱུ་ལ་གནས་པ་དང་། །ངན་སོང་འགྲོ་བའི་ལས་ནི་བྱེད་པར་འགྱུར། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །མི་མཛེས་པ་དང་བལྟ་ན་མི་སྡུག་དང་། །དེ་ནི་རྟག་ཏུ་རུས་རིགས་དམའ་བར་འགྱུར། །དེ་ཡི་ལྕེ་ཡི་དབང་པོ་རྩུབ་འགྱུར་ཏེ། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །ལུང་མནོས་ཆོས་རྣམས་ལེགས་པར་མི་གནས་ཤིང་། །ལུང་མནོས་མ་བྱས་མདོ་སྡེ་ཐོས་མི་འགྱུར། །དགེ་བའི་བཤེས་གཉེན་རྣམས་ཀྱང་རྣམ་པར་སྤོང་། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །སྡིག་པའི་གྲོགས་པོ་དག་དང་ཕྲད་པར་འགྱུར། །དཀའ་བའི་ཐེག་པས་རྣམ་པར་དག་འགྱུར་ཞིང་། །མི་འདོད་པ་ཡི་ཚིག་རྣམས་རྒྱུན་དུ་ཐོས། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །བྱིས་པའི་བློ་ཅན་དེ་ནི་གར་སྐྱེས་དེར། །ཡིན་ནམ་མ་ཡིན་སྙམ་དུ་དེ་རྟོག་ཅིང་། །རྟག་ཏུ་དེས་ནི་ཐེ་ཚོམ་རྙེད་པར་འགྱུར། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །དེ་ནི་མི་ཁོམ་བརྒྱད་དང་རིང་མི་འགྱུར། །དེ་ནི་དལ་བ་ལྡན་པའི་མཆོག་མི་འགྱུར། །དེ་ལ་གནོད་པ་འདི་དག་རྟག་འབྱུང་སྟེ། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །དེ་ཡི་དགེ་བ་དག་ལ་བར་ཆད་མང་། །བགེགས་སུ་འགྱུར་བས་མྱུར་དུ་འཐོབ་མི་འགྱུར། །བརྩོན་དང་ལྡན་ཡང་དེ་ལ་དགྲ་རྣམས་འབྱུང་། །སྤྲོས་སྤྱོད་དེ་ཡི་ཉེས་པ་འདི་དག་གོ །མཁས་པས་ཉེས་འདི་ཡང་དག་རིག་ནས་ནི། །སྤྲོས་པ་ཐམས་ཅད་ཡོངས་སུ་སྤང་བར་བྱ། །སྤྲོས་སྤྱོད་དག་ནི་གནོད་པ་རྙེད་པར་སླ། །དེ་ལྟ་བས་ན་སྤྲོས་ལ་གནས་མི་བྱ། །གང་ལ་སྤྲོས་པའམ་རྩོད་པ་ཡོད་པ་ལས། །དཔག་ཚད་བརྒྱ་ཡི་ཕ་རོལ་འགྲོ་བ་མཆོག །གང་ན་ཉོན་མོངས་ཡོད་པ་དེ་དག་ཏུ། །ཡུད་ཙམ་ཞིག་ཀྱང་གནས་ཤིང་འདུག་མི་བྱ། །རབ་བྱུང་དོན་འདོད་ཡོན་ཏན་འདོད་པ་རྣམས། །སྡང་བའི་སེམས་ཀྱིས་རྩོད་པར་མ་བྱེད་ཅིག །ཁྱེད་ལ་ཞིང་མེད་ཞིང་ལས་ཚོང་བྱར་མེད། །གང་ལ་ནོར་ཕྱིར་འདི་དག་འབྱུང་བ་སྤྲོས། །ཁྱེད་ལ་ཆུང་མ་བུ་དང་བུ་མོ་མེད། །ཁྱིམ་དང་གཉེན་
གྱི་ཚོགས་ཀྱང་ཡོད་མ་ཡིན། །བྲན་ཕོ་བྲན་མོ་མེད་ཅིང་དབང་ཕྱུག་མེད། །རབ་ཏུ་བྱུང་ནས་རྩོད་པར་མ་བྱེད་ཅིག །ཞི་ཞིང་རབ་ཏུ་ཞི་བས་བསྟེན་པ་ཡི། །ངུར་སྨྲིག་གོས་རྣམས་དད་པས་བླངས་ནས་ནི། །ཞི་ཞིང་རབ་ཞི་ཉེ་བར་ཞི་བར་བྱོས། །སྤྲོས་པ་རྣམ་པར་སྤོངས་ལ་བཟོད་པ་སྐྱེད། །མི་བཟད་སེམས་ནི་སྦྲུལ་དུག་འདྲ་བ་སྲུངས། །སྤྲོས་ལ་སྤྱོད་པ་སེམས་ཅན་དམྱལ་བ་དང་། །དུད་འགྲོ་གཤིན་རྗེའི་ཡུལ་དང་རིང་མི་འགྱུར། །དེ་ལྟ་བས་ན་བཟོད་དང་བརྩོན་འགྲུས་སྐྱེད། །ཚོགས་པ་གཞིག་ཕྱིར་སྤྲོས་པར་བྱེད་ན་ནི། །སྡུག་བསྔལ་གསོད་དང་འཆིང་བ་ཅི་ཡང་རུང་། །གོ་ར་བརྡེག་དང་བསྡིགས་པ་དེ་དག་ནི། །འཇིག་རྟེན་འདི་ན་གནས་པ་རྣམས་ལ་འབྱུང་། །ཚོགས་པར་གྱུར་ལ་ཉེས་པ་འདི་དག་དཀོན། །ཚོགས་གྱུར་བག་ལ་ཉལ་བ་འཕེལ་བ་མེད། །ཚོགས་གྱུར་རྣམས་ལ་དགའ་བ་འབྱུང་འགྱུར་ན། །མཁས་པ་སུ་ཞིག་ཚོགས་པ་སྐྱེད་མི་བྱེད། །གླགས་ཚོལ་དག་གིས་སྐྱོན་རྣམས་རྙེད་མི་འགྱུར། །ནམས་ཀྱང་འཁོར་རྣམས་ཕྱེད་པར་མི་འགྱུར་ཏེ། །དེ་ཡི་མཛའ་བཤེས་དག་ཀྱང་རྒྱུན་དུ་འབྱུང་། །སྤྲོས་པ་སྤངས་པའི་ཕན་ཡོན་དེ་དག་ཡིན། །ཐེག་པའི་སྟོབས་འདི་ཡིས་ནི་དག་འགྱུར་ཞིང་། །ལས་ཀྱི་སྒྲིབ་པ་མ་ལུས་བྱང་ནས་ནི། །སྟོབས་དང་བཅས་ཤིང་བཞོན་བཅས་བདུད་ཐུལ་ཅིག །བརྟན་པ་གང་ཡིན་དེས་ནི་བཟོད་པ་བསྐྱེད། །སྤྲོས་པ་རྒྱས་པ་ལ་ནི་སྐྱོན་མང་ཆགས། །ཉེས་པ་དེ་དག་ངས་ནི་བསྟན་མི་ནུས། །རྟག་ཏུ་སྤྲོས་མེད་བསྔགས་པ་མཐའ་ཡས་འགྱུར། །སུ་ཞིག་དོན་གཉེར་དེས་ནི་བཟོད་པ་སྐྱེད། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་ཀུན་ནས་ཉོན་མོངས་པ་དེ་ཙམ་དུ་གླེང་བ་ནི་དེ་བཞིན་གཤེགས་པས་ལེགས་པར་གསུངས་ཏེ་ངོ་མཚར་ལགས་སོ། །བཅོམ་ལྡན་འདས་ཅི་ལྔ་བརྒྱ་པ་ཐ་མ་ལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་ཚིག་གི་མཆོག་འདི་ལྟ་བུ་འདི་དག་ཐོས་ན་སྐྱོ་བར་འགྱུར་ཞིང་ཀུན་ནས་ཉོན་མོངས་པའི་སྤྱོད་པ་ཡང་སྤོང་བར་འགྱུར་ལགས་སམ་བཀའ་སྩལ་པ། བྱམས་པ་ལྔ་བརྒྱ་པ་ཐ་མ་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཐེག་པ་པའི་གང་ཟག་རྣམས་ཚིག་གི་མཆོག་འདི་ལྟ་བུ་འདི་དག་ཐོས་ན་སྐྱོ་བར་འགྱུར་ཞིང་ཀུན་ནས་ཉོན་མོངས་པའི་སྤྱོད་པ་ཡང་སྤོང་བར་འགྱུར་བ་འབྱུང་བ་དེ་དག་ནི་ཉུང་ངོ་། །བྱམས་པ་གང་སེམས་ཀྱི་རྒྱུད་ཁེངས་པར་འགྱུར་ཞིང་གུས་པ་མེད་པ་དང་། མངོན་པའི་ང་རྒྱལ་ལ་གནས་པ་དང་། གཅིག་ལ་གཅིག་ཐེ་ཚོམ་ཟ་བ་མང་བ་དག་ནི་ཕལ་ཆེ་སྟེ། དེ་དག་ནི་དེ་བཞིན་གཤེགས་པས་གསུངས་པའི་མདོ་སྡེ་ཟབ་མོ་བསྟན་པའི་ཡོན་ཏན་དང་ཕན་ཡོན་ཟབ་པ་དག་ལུང་ནོད་པར་མི་བྱེད། ཁ་ཏོན་དུ་མི་བྱེད། འཆང་བར་མི་བྱེད། ཀུན་ཆུབ་པར་མི་བྱེད་དེ། དེ་དག་གི་ལས་ཀྱི་སྒྲིབ་པ་ཤས་ཆེ་བས་ཡོན་ཏན་དེ་དག་དང་ཕན་ཡོན་དེ་དག་འབྱུང་བར་མི་འགྱུར་རོ། །དེ་དག་དད་པ་མེད་པ་དེས་མདོ་སྡེ་དེ་དག་ལ་ཐེ་ཚོམ་ཟ་བར་འགྱུར་ཏེ། ཕྱིས་ཀྱང་ལུང་ནོད་པར་མི་བྱེད། འཆང་བར་མི་བྱེད། ཀུན་ཆུབ་པར་མི་བྱེད་དེ། དེ་དག་གི་དྲུང་དུ་བདུད་སྡིག་ཅན་དགེ་སློང་གི་ཆ་ལུགས་སུ་འདུག་ནས་འདི་སྐད་དུ་མདོ་སྡེ་འདི་དག་ནི་རང་བཟོར་བྱས་པ་ཡིན་ཏེ། འདི་དག་ནི་དེ་བཞིན་གཤེགས་པས་གསུངས་པ་མ་ཡིན་ནོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། མདོ་སྡེ་འདི་དག་ལས་ཡོན་ཏན་དང་ཕན་ཡོན་བཤད་པ་གང་ཡིན་པ་དེ་དག་ཁྱེད་ཀྱི་འདི་ལས་
མི་འབྱུང་ཞིང་མེད་པའི་ཕྱིར་རོ་ཞེས་མོས་པ་ཟློག་པར་བྱེད་དོ། །དེ་དག་དེ་ལྟར་མོས་པ་ཟློག་པས་མདོ་སྡེ་ཟབ་མོ་དག་མི་འདོད་ཅིང་ཐེ་ཚོམ་དུ་འགྱུར་ཏེ། ལུང་ནོད་པར་མི་བྱེད། ཁ་ཏོན་དུ་མི་བྱེད་དེ། མི་བླུན་པོ་དེ་དག་གིས་ཡོན་ཏན་དང་ཕན་ཡོན་དེ་དག་ཐོས་ན་འདི་ལྟར་ཆོས་སྤོང་བར་འགྱུར་བའི་ལས་ཀྱི་རྣམ་པར་སྨིན་པ་ནི་སྦྱང་བར་མི་ནུས་སོ་སྙམ་དུ་མི་སེམས་སོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཚེ་དཔག་མེད་ཀྱི་ཡོན་ཏན་དེ་བཞིན་གཤེགས་པས་གསུངས་པ་གང་ལགས་པ་འདི་དག་ལས་གང་གིས་སེམས་བསྐྱེད་པ་གང་ཡང་རུང་བ་ཞིག་ཡིད་ལ་བགྱིད་ཅིང་། དེ་བཞིན་གཤེགས་པ་ཚེ་དཔག་མེད་ཡིད་ལ་བགྱིད་པ་དེ་སངས་རྒྱས་ཀྱི་ཞིང་དེར་སྐྱེ་བར་འགྱུར་ན། བཅོམ་ལྡན་འདས་སེམས་བསྐྱེད་པ་དེ་དག་གང་ལགས། དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །བྱམས་པ་སེམས་བསྐྱེད་པ་གང་དག་གིས་དེ་བཞིན་གཤེགས་པ་ཚེ་དཔག་མེད་ཀྱི་སངས་རྒྱས་ཀྱི་ཞིང་འཇིག་རྟེན་གྱི་ཁམས་བདེ་བ་ཅན་དུ་སྐྱེ་བར་འགྱུར་བར་བྱེད་པ་དེ་དག་ནི་བྱིས་པའི་སེམས་བསྐྱེད་པ་མ་ཡིན། སྐྱེས་བུ་དམ་པ་མ་ཡིན་པའི་སེམས་བསྐྱེད་པ་མ་ཡིན། ཀུན་ནས་ཉོན་མོངས་པའི་སེམས་བསྐྱེད་པ་མ་ཡིན་ཏེ། བྱམས་པ་སེམས་བསྐྱེད་པ་ནི་བཅུ་པོ་འདི་དག་ཡིན་ནོ། །བཅུ་གང་ཞེ་ན། བྱམས་པ་འདི་ལྟ་སྟེ། སེམས་ཅན་ཐམས་ཅད་ཀྱི་སྤྱོད་པ་ལ་ཞེ་འགྲས་པས་ནི་དེ་བཞིན་གཤེགས་པ་ཚེ་དཔག་མེད་ཀྱི་སངས་རྒྱས་ཀྱི་ཞིང་དུ་སྐྱེ་བར་མི་ནུས་ཏེ། སེམས་ཅན་ཐམས་ཅད་ལ་བྱམས་པའི་སེམས་བསྐྱེད་པ་དང་། འཚེ་བ་ཡོངས་སུ་སྤངས་ཏེ། སེམས་ཅན་ཐམས་ཅད་ལ་སྙིང་རྗེ་བའི་སེམས་བསྐྱེད་པ་དང་། ལུས་དང་སྲོག་ལ་མི་ལྟ་བར་དམ་པའི་ཆོས་བསྲུང་བའི་སེམས་བསྐྱེད་པ་དང་། ཟབ་མོའི་བཟོད་པ་ལ་ངེས་པར་གྱུར་པས་དམ་པའི་ཆོས་ལ་མངོན་པར་ཞེན་པའི་སེམས་བསྐྱེད་པ་དང་། རྙེད་པ་དང་བཀུར་སྟིས་མ་གོས་པས་བསམ་པ་ཡོངས་སུ་དག་པའི་སེམས་བསྐྱེད་པ་དང་། བརྗེད་པ་མེད་པས་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ལ་རིན་པོ་ཆེར་སེམས་བསྐྱེད་པ་དང་། ཞུམ་པ་མེད་པས་སེམས་ཅན་ཐམས་ཅད་ལ་བླ་མ་བཞིན་དུ་གུས་པའི་སེམས་བསྐྱེད་པ་དང་།བྱང་ཆུབ་ཀྱི་ཡན་ལག་ལ་ངེས་པས་ཉི་ཚེ་བའི་བསམ་གཏན་གྱི་རོ་མྱང་བར་མི་བྱ་བའི་སེམས་བསྐྱེད་པ་དང་། མ་འདྲེས་པས་དགེ་བའི་རྩ་བ་རྣམ་པ་སྣ་ཚོགས་ཀྱི་སེམས་བསྐྱེད་པ་དང་། མཚན་མ་ཡོངས་སུ་སྤངས་ཏེ། སངས་རྒྱས་རྗེས་སུ་དྲན་པ་བསྒོམ་པའི་སེམས་བསྐྱེད་པ་སྟེ། བྱམས་པ་སེམས་བསྐྱེད་པ་བཅུ་པོ་ནི་དེ་དག་ཡིན་ཏེ། སེམས་བསྐྱེད་པ་བཅུ་པོ་དེ་དག་ལས་སེམས་བསྐྱེད་པ་གང་ཡང་རུང་བའི་སེམས་བསྐྱེད་པ་གང་ཡིན་པ་དེ་ནི་སེམས་བསྐྱེད་པ་དེ་དག་དང་ལྡན་ན་སངས་རྒྱས་ཀྱི་ཞིང་དེར་སྐྱེ་བར་མི་འགྱུར་བ་དེ་ནི་གནས་མེད་དོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་
