[𰶞48a]

པ་ཡང་ཉེ་བར་སྒྲུབ་པར་བྱེད་དོ། །ནི་རཱ་ལཾ་བྷེ། ཨ་གྲ་ཀ་རེ། ཀྵེ་མ་བི་ཙ་ཡེ། ནིར་བྷ་ས་བ་ཏི། ནིར་ཧ་ར་བ་ཏི། ཨོ་ཛོ་བ་ཏི། ཨུ་པ་ན་མ་ཏི། ཨུ་པ་སཾ་ཧ་ར་ཀ་ར་ཎི། ཨ་བྷི་ཤ་ཐ་ཨི་ཧ་དྷཱ་ར་ཎི་མུ་ཁེ། དྷརྨ་མུ་ཁེ། དྷརྨ་པ་ཏ་ལེ། བཀོད་པ་མཐའ་ཡས་ལྷ་འདི་དག་ནི་རིའི་རྒྱལ་པོ་གངས་ཅན་གྱི་ལྷོ་ཕྱོགས་ཀྱི་ངོས་ལ་གནས་ཤིང་། གང་དག་ཆོས་འདི་ལ་བརྩོན་པར་འགྱུར་བའི་ཆོས་སྨྲ་བ་དེ་དག་ལ་གཟི་མདངས་ཉེ་བར་སྒྲུབ་པར་བྱེད་དོ། དྷརྨ་ཀཱ་མ། དྷརྨ་ཙན་ཏི་ཀ། ཀ་ལྱ་ཎཱརྠ་བི་ས་རྫ་ནི།ཀཽ་ཤ་ལྱ་ཨ་ནུ་ག་ཏི། ཨུ་པཱ་ཡ་སཾ་གྲི་ཧཱི་ཏེ། བི་ནི་ར་མུག་ཏེ། ཤཱན་ཏ་པ་ད་བི་ས་ར་ཎི། ཨོ་བྷཱ་ས་ཡ་ཤ་བ་ཏི། བཀོད་པ་མཐའ་ཡས་ལྷ་འདི་དག་ནི་རྒྱ་མཚོ་ཆེན་པོའི་ངོགས་ན་གནས་ཤིང་ཆོས་སྨྲ་བ་དེ་དག་ལ་བསོད་སྙོམས་ཉེ་བར་སྒྲུབ་པར་བྱེད་དེ། དེ་བཞིན་གཤེགས་པ་ནི་གང་དག་མདོ་འདི་ཉན་པ་དེ་དག་གི་དོན་དུ། གསང་སྔགས་ཀྱི་བརྡ་འདི་སྟོན་པར་མཛད་དོ། །གསང་སྔགས་འདི་དག་ནི་ལྷ་རྣམས་ཀྱི་དབང་པོ་བརྒྱ་བྱིན་འགུགས་པར་འགྱུར་བ་ཡིན་ཏེ། དེ་དག་ཀྱང་གང་ཞེ་ན། མ་ཁི་པ་ཏི། བ་སུ་མ་ཏི། དེ་བ་རཱ་ཛཱ་ཤ་ཙི་མ་པ་ཏི། སརྦ་ཨ་སུ་རཱ་ཎཱ་ན་ནིར་གྷ་ཏ་ནི། བཛྲེ་ཡ་སུ་པྲ་ཏིཥྛི་ཏ། པྲ་མརྡྷ་ན་ཨུ་པ་ཨ་སུ་རཱ་ཎཱཾ། དེ་བ་ན་ཨ་དྷི་པ་ཏི་མ་སཱ་དྷ། པུ་ར་སྐྲི་ཏོ་པི་དེ་བེ་ཧི། ཤོ་བ་སེ་བྷ་ས་བ་ས་དྷ། པུ་རན་ད་ར་མ་ཧཱ་ཀེ་ཏུ་ཏཱ་པེ་སི། ཨ་སུ་ས་མ་ད། དྷྭ་ཛ་གྲན་ཏ། པཱ་པ་ཤྱིན་ཏཱ་བི་ངྷྱཾ་སི་ཏ་མ་ཧཱ་ཨ་སུ་ར་ཨིན་དྲས་པ་ལ་བན་ཏི་བྷི་ཏ་ཏྲ་སྟ་དི་ཤོ་དི་ཤ། ནམ་ན་བུད་དྷ་སྱ་ཀཱིར་ཏི་ཤྱེ། ནཱ་མ་གོ་ཏྲན་མ་ཧཱ་ཡ་ཤ། དེ་བ་པུ་ར་སྨི་སཾ་གྲ་མེ། བུད་དྷ་ཤ་བ་ཏ་ཨ་ནུད་ཏ་རཾ་ཤྲཱ་པི་སི་དེ་བ་བི་ཛ་ཡ། ན་མོ་བུད་དྷ་པི་བྷཱ་ཤིཏ། སཾ་ཀྲཱ་མ་ཚན་ད་སི་བིདྱཱ་བུད་དྷ་ནཱ་མེན་བྷ་ཧི་ཏཱ། བི་ཛ་ཡེ་སོ་ཏ་ཏ། དེ་བ་ཨ་སུ་ར་ཤྩ བི་ནི་ཏ་བན། ཀྲི་ཏཱ་ཏེ། རཀྵ་དེ་བ་ནཱཾ། མ་ནུ་ཥྱ་ནཱཾ། རཱ་ཀྵ་པི་ཏ། ཡཾ་ཨི་ཧ་དྷརྨ་ནིར་དེ་ཤ། ཨོ་ཏ་རན་ཏི་སུ་པྲ་ཏིཥྛི་ཏ། ནི་གྲི་ཧི་ཏ་དྭ་ཡཱ། ཨ་སུ་ར། རཱཀྵ་ས། ཡ་ཀྵ། ཀིན་ན་ར། ནཱ་ག། ཀུམ་བྷན་ཌ། བྷུ་ཧཱ་ནི། བཱུ་ཏ་ནི། པི་ཤཱ་ཙ། དཱན་བཱ་ཏ་ཐཱ། ཨ་ཙ་ལ་ཨསྠཱ་ན་བྲཱབ་ཏོ་སི། ཨིན་དྲ་ཨིན་དྲ་པུར་སྐྲི་ཏ། ཨ་ཏི་གམ་བྷཱི་ར། ཤཱ་ན་ཏ་ན་ཙ། ཨ་གྲ་པུ་རུ་ཥ་སེ་བི་ཏ། མུ་དྲ་བན་ཏོ་སི། དེ་བ་ཨིན་དྲ། བུད་དྷི་མན་ཏ་ཐཱ་བི་ཤུད་དྷ། ཨ་བི་ཤ། །ཨི་ཧ་སཱུ་ཏྲ་སྨི་རཀྵ་སམ་བི་ཏ་ཧི་ཙ། ས་ཧ་སྲ་ནེ་ཏྲ་སུ་མུ་ཁ་ཨ་བ་ས་ར་པ་རི་བརྟ། ཨུ་པ་བྷུཉྫ་སྭ་ཀ་ཀརྨ་པུརྦ་ཧེ་ཏུ་སྨི་ཡང་ཀྲི་ཏམ། ཨ་ཏི་པ་ཤྩ་ཨནུ་བྲཱ་པ་ཏ། ཨ་ཙ་ལ་སུ་པྲ་ཏིཥྛི་ཏ། ཀྲི་ཏཾ་ཧི་ཀུ་ཤ་ལང་བཱུ་རེ། མ་ནུ་ཥྱེ་ཤུ་མ་ཧི་པ་ཏི། པཱུ་རྦཾ་ག་མ་ཨུ་སི་དེ་བ་ནཱཾ། པཱུ་ཛེན་ཏི་ན་ཙ་མ་ནུ་ཥ། བུད་དྷ་སྱ་ཀྲིཏ་ཏྭ་སད་ཀཱ་རཾ། པཱུ་ཛ་བེ་ཤྱ་ཏི་ཏེ་བ་ཧུ། མ་ཧཱ་ཙ་མ་ནི། བི་བ་རབྷ་བན་མ་བ་ཤ་ཨུད་བ་ཏེ། དེ་བཱལ་ཡ་སྱ་མ་དྷ་སྨི་ཡ་ཀྵོ་ཧི་པ་རི་བཱ་ར་ཏོ། །པྲ་ལཾ་བ་ཧ་རི་བཧུ་ཏ། བསྐ་བི་ཤ་ཏ། ཨ་གྲ་བ་ཧི་ས་ཏི་མ་ཏི། སུ་ར་ཏ་པ་ཏི། སུ་ར་ཏ་ཐཱ་པྲ་བྷུ་སྐན་དྷ། ཨ་ནི་ལ་ཙ་མ་རཱད་ཙྪ་མ་ཧཱ་བྷཱ་གི་མ་ཧཱ་སྐན་དྷ། མ་ཧཱ་ཧེ་ཏུ་སུ་པྲ་དྷི་པ། མ་ཧ་བ་ལམ། ཨ་ཏི་ཡཀྵ་མ་ཧཱ་ཏེ་ཛ། རཀྵ་ན་ཏི་བྷ་བཱ་ནན་ཏ་པ། བ་ཧུ་ཡཀྵ་པ་རི་བཱ་རོ། །ཡེ་ན་ཤོ་བྷ་སི་པཱ་ས་ཙ། བ་སུ་སརྦ་བ་སུ་ཏ་མ་ཧཱ་ཏེ་ཛ། མ་ཧཱ་ག་ནི། མ་ཧཱ་པྲ་ཧ་ར་ནོ་ཙ་ཡི་བ་ཨ་བྷི་ནིར་ཧ་ར་ཡཱ་ཏ་ཐཻ་བ་ར་ཙ། སུ་མེ་རུ་མུར་ནི་ཏ་པ། བཱ་ས་བ་སརྦ་དེ་བ་པུ་ར་སྐྲི་ཏོ། །ཤོ་བྷ་ཏེ་བྷ་བ་ན། ཨོ་བྷ་སཾ་པྲ་བྷ་མུཉྩ་སི་ཏེ་ཛ་ས། འདི་དག་ནི་ལྷ་རྣམས་ཀྱི་དབང་པོ་བརྒྱ་བྱིན་དབབ་པའི་ཚིག་ཡིན་ཏེ། ཚིག་དེ་དག་གིས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་འགུགས་པར་འགྱུར་རོ། །དེ་ལ་འདི་དག་མ་འོངས་པའི་དུས་ན་སེམས་ཅན་གང་དག་བྱང་ཆུབ་ལ་ཡང་དག་པར་ཞུགས་ཏེ། ཆོས་རྣམས་ཡང་དག་པར་འཛིན་ཅིང་། སེམས་ཅན་རྣམས་ལ་ཆོས་ཡང་དག་པར་བསྡུ་བ་ལ་མཁས་པ་ཉེ་བར་གནས་པར་འགྱུར་བའི་གསང་སྔགས་ཀྱི་ཚིག་ཡིན་ནོ། །ལུས་ངན་པོ་ཡི་བུ་ཐུ་བོ། །གནོད་སྦྱིན་ཡང་དག་ཤེས་ཞེས་བྱ། །ཁྲོ་མེད་རྩོལ་བ་མེད་གནས་ཏེ། །ཕ་དང་བཅས་ཏེ་རྟག་ཏུ་ནི། །ཆོས་འདི་
དག་ལ་དགའ་བ་གང་། །དེ་ལ་སྲུང་བ་བྱེད་པར་འགྱུར། །དེ་བཞིན་རྒྱལ་པོ་ཡུལ་འཁོར་སྲུང་། །འཁོར་དང་བཅས་པས་རྟག་ཏུ་ནི། །ལེགས་པར་བཤད་པ་མདོ་ཡི་སྡེ། །དེ་ལ་སྲུང་བ་བྱེད་པར་འགྱུར། །མིག་མི་བཟང་ཞེས་བྱ་བ་ནི། །རང་གི་འཁོར་བཅས་སྟོབས་དང་ལྡན།ཚུལ་འདི་ལ་ནི་གང་གནས་པ། །དེ་དག་ཐམས་ཅད་སྲུང་བར་འགྱུར། །དེ་བཞིན་རྒྱལ་པོ་འཕགས་སྐྱེས་པོ། །བཞོན་བཅས་སྟོབས་དང་ལྡན་པ་སྟེ། །ཆོས་འདི་དག་ལ་འདོད་པ་གང་། །དེ་དག་ཀུན་ཏུ་སྲུང་བར་འགྱུར། །ས་ནི་བེ་ཤ་ནི། མ་ཧཱ་ས་རི། མ་ཧཱ་ག་ནེ། མ་ཧཱ་ཛ་ནི། དྲ་བྷུ་ཏ་བི་ཛ་ཡེ། དྷྭ་ཛ་ཨ་གྲ། ཨུ་པ་བྷ་ས་ནི། ཨ་ནི་ལ་སེ་ན་ཏུ་ས་ཧ། ནཱ་ནཱ་པར་ཎ་ད་ཤ་དི་ཤ། །ཙད་ཏྭ་ར་ལོ་ཀ་པཱ་ལ་ནི། ཏུསྨེ་རཱ་ཛཱ་ན་ཨུད་ཙྱ་ཐཱ། ཨཱ་བེ་ཤ་ན། ཨི་ཧ་ས་མན་ཏ་ཧ་ར་ཐ། པྲ་ཛེ་ནི་གྲ་ཧི་ས་ར་བེ་ཙ་ཏུ་ར་དི་ཤ། །བཀོད་པ་མཐའ་ཡས་འདི་ནི་རྒྱལ་པོ་ཆེན་པོ་བཞི་འཁོར་དང་བཅས། གཡོག་དང་བཅས། བུ་དང་བཅས། ཆུང་མ་དང་བཅས། རྗེས་སུ་འབྲང་བར་བཅས་པ་རྣམས་དབབ་པའི་རིག་པ་ཡིན་ནོ། །ཤར་གྱི་ཕྱོགས་ན་གནས་པ་ཡི། །སྲིན་མོ་རྒྱལ་མཚན་ཏོག་དང་ནི། །ཏོག་ཆེན་དང་ནི་སྟོབས་ཆེན་མོ། །གྲགས་པ་ཆེན་པོ་གང་ཡིན་པ། །དེ་དག་ཀུན་ཏུ་འབེབས་འགྱུར་ཞིང་། །དེ་དག་ཐམས་ཅད་འཁོར་བཅས་ཏེ། །རྟག་ཏུ་ཆོས་ཀྱི་སྒོ་འདི་ལ། །སྲུང་བ་བྱེད་ཅིང་ཉེ་བར་གནས། །རབ་འཕྱང་རྣམ་པར་འཚེར་བ་དང་། །དོན་གྲུབ་མ་དང་འཕྲོག་མའི་བློ། །འདི་དག་ཐམས་ཅད་ཉེ་བར་ནི། །ལྷོ་ཡི་ཕྱོགས་ན་གནས་པ་ཡིན། །དེ་དག་ཡང་དག་འདིར་བབས་ནས། །གཟི་མདངས་ཉེ་བར་སྒྲུབ་བྱེད་ཅིང་། །སྲུང་བའི་མཆོག་ཀྱང་བྱེད་འགྱུར་བར། །ཐམས་ཅད་གཟིགས་པས་ངེས་པར་བསྟན། །སྲིན་མོ་ལ་བ་ཅན་དང་ནི། །སྡོམ་པ་དང་ནི་སྒྲ་སྒྲོགས་དང་། །འདྲེས་པ་གྲགས་དང་ལྡན་པ་གང་། །ནུབ་ཀྱི་ཕྱོགས་ན་གནས་པ་སྟེ། །དེ་དག་ལྷན་ཅིག་ཀུན་བབས་ནས། །ཆོས་འདིའི་དོན་ཤེས་གང་ཡིན་པ། །དེ་ལ་ཆོས་བསྟན་བྱ་བའི་ཕྱིར། །རྟག་ཏུ་སྲུང་བར་བྱེད་པར་འགྱུར། །སྒོ་འདི་ལ་ནི་མ་དད་དང་། །དེ་བཞིན་བདེན་དང་བདེན་མི་ལྡན། །སྲིན་མོ་གྲགས་ཆེན་དེ་དག་ནི། །བྱང་གི་ཕྱོགས་ན་གནས་པ་སྟེ། །དེ་རྣམས་ཡང་དག་འདིར་བབས་ནས། །དེ་བཞིན་གཤེགས་པའི་གཟི་བརྗིད་ཀྱིས། །མི་མཆོག་བསྲུང་བར་བྱ་བའི་ཕྱིར། །ཐམས་ཅད་ཐལ་མོ་སྦྱར་ཏེ་འཁོད། །བཀོད་པ་མཐའ་ཡས་དེ་ལ་བདུད་སྡིག་ཅན་ཚར་གཅད་པའི་ཚིག་དང་། ཚངས་པ་དབབ་པའི་ཚིག་གང་ཡིན་པ་དེ་དག་བསྟན་པར་བྱས། ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག །མི་ཏྲེ། མི་ཏྲ་བ་ཏི། ཀུ་རུ་ཎེ། ཀཱ་རུ་ཎ་བ་ཏི། བི་བུད་དྷི། བི་བུད་དྷ་བ་ཏི། པྲ་མོ་ཙ་ནི། པྲ་མོ་ཙ་ན་བ་ཏི། ཀྲི་ཏ། ཀྲི་ཏ་བ་ཏི། ཨ་ནུ་ག་མི། ཨ་ནུ་ག་མ་བ་ཏི། ཨུ་པད་ཚེ་ད་ནི། ཀ་མ་བི་བརྫ་ནི། ཀྲིཥྞ་ས་མུད་ཤོ་ཥ་ཎི། ནི་གྲི་ཧཱི་ཏ་མཱ་ར་བ་ལཾ། ཨུད་ཏ་ར་ཎ་བ་ཏི། པྲ་ཏྱུད་ཏཱ་ར་ཎ་བ་ཏི། ཨུ་པེ་ཀྵ་ཨ་ནུ་ས་ཧི་ཏེ། ཨཱ་རམ་པ་ཎ་བི་ཤོ་དྷ་ནི། ནི་དར་ཤ་ན། ཨ་ས་མོ་ཧ་ནི། ནི་སྤྲ་པཉྩ་བ་ནེ། བཀོད་པ་མཐའ་ཡས་འདི་དག་ནི་བདུད་སྡིག་ཅན་ཚར་གཅད་པའི་ཚིག་ཡིན་ཏེ། དེ་དག་གིས་བདུད་སྡིག་ཅན་དཔུང་དང་བཅས་ཤིང་བཞོན་པ་དང་བཅས་པས་གླགས་རྙེད་པར་མི་འགྱུར་རོ། །དེ་ལ་ཚངས་པ་ཆེན་པོ་དབབ་པའི་ཚིག་གང་ཡིན་ཞེ་ན། ཨ་ཏི་པ་ཏི་མ་ཧཱ་སྠཱ་ན་པ་ཏི། སྭ་ཡ་ཀྲི་ཏ་སྠཱ་ན་མ་ཏི། ནཱ་ནཱ་བ་ཧུ་བི་བི་དྷ། པྲ་ཏྱུད་པསྠན་ཀ་ར་ཎ།ཨ་དྷི་ཀ་སྠཱ་ན་པ་ཏི། ཤུད་དྷ་སྠཱ་ན་ག་ཏོ། །ས་ཧ་མ་པ་ཏི། ཨ་དྷི་ཀྲཱན་ཏ་བི་ཤོ་ཥ་ཎ་ཤུ་བྷ་བ་རི། བར་ཤུ་བྷ་ཨ་དྷམུ་ག་ཏ། བྲཱ་པ་ཏ་ཤྲི་ཏ། ནི་བཱ་སན་པྲ་བྷུ་ཏ་པྲ་བྷ། ནི་གྲ་ཧཱི་ཏ་ཀཱ་མ་མ། སྭ་སྟི་ཀ་མ་མ་པྲ་ཏྱུ་པ་སྠཱ་ན། འདི་དག་ནི་ཚངས་པ་ཆེན་པོ་དབབ་པའི་ཚིག་ཡིན་ཏེ། དབབ་པའི་ཚིག་འདི་དག་གིས་ཚངས་པ་ཆེན་པོ་ཆོས་སྨྲ་བ་དེ་དག་ལ་ཚིག་དང་ཡི་གེ་ཡོངས་སུ་དག་པ་དང་། ཚངས་པར་སྤྱོད་པ་ཤིན་ཏུ་མཉམ་པར་གཞག་པ་ཡོངས་སུ་རྫོགས་པ་ཉེ་བར་སྒྲུབ་པར་བྱེད་དོ། །བི་ཤོ་དྷ་ན་བ་ཏི། ཨན་ཏི་མ་ཧ་ཀཱ་རུ་ཎི་ཀ། ཀཱ་མ་བྷ་བ་ཨུ་ས་ཏ་ནི། བརྱ་ནི་ར་མི་ཏ་ནི་པ་རྱན་ཏ་པྲ་ཏྱུ་པ་སྠཱ་ན། པརྱ་བ་སཱ་ན། ཨཱ་ལ་ཡ་བི་ཤོ་དྷ་ནི། ཨཱད་ཛྙཱ་པརྱན་ཏེ། ཀ་ར་ཎི་ཨན་ཏི་མ། དེ་ཤ་པྲ་ཏྱུཔ་སྠཱ་ནེ། བཀོད་པ་མཐའ་ཡས་འདི་དག་ནི་གནས་གཙང་མ་པའི་རིས་ཀྱི་ལྷ་རྣམས་ཀྱི་ཆོས་སྣང་བའི་ཚིག་ཡིན་ཏེ། ཆོས་སྣང་བའི་ཚིག་དེ་དག་གིས་རིགས་ཀྱི་བུ་དེ་དག་ལ་ཆོས་ཀྱི་གཞི་འདི་ཉེ་བར་སྒྲུབ་པར་འགྱུར་རོ། །བཀོད་པ་མཐའ་ཡས་དེ་ལ་ལྷའི་དབང་པོ་དང་། ཀླུའི་དབང་པོ་དང་། ལྷ་མ་ཡིན་གྱི་དབང་པོ་དང་། མི་འམ་ཅིའི་དབང་པོ་དང་། ནམ་མཁའ་ལྡིང་གི་དབང་པོ་དང་། སེམས་ཅན་གང་དག་མཐུ་ཆེ་བའམ། མཐུ་ཆུང་བ་འང་རུང་། དད་པའམ་མ་དད་པ་ཡང་རུང་སྟེ། དེ་དག་གི་གསང་སྔགས་ཀྱི་ཚིག་འདི་ཉེ་བར་བསྒྲུབ་པར་བྱའོ། །དེས་ན་སེམས་ཅན་དད་པ་རྣམས་ནི་ཆོས་བསྟན་པ་འདི་ལ་མཆོག་ཏུ་འཇུག་པར་འགྱུར་ལ། སེམས་ཅན་མ་དད་པ་དེ་དག་ཀྱང་ཅང་མི་སྨྲ་བ་དང་། བཏང་སྙོམས་སུ་འཇོག་པར་འགྱུར་གྱི། རྩོད་པ་དང་འགྱེད་པར་ཡང་མི་འགྱུར་ལ། ཆོས་ཀྱི་སྒོ་འདི་ཐ་མལ་པར་འཇོག་གི། གནོད་པར་བྱ་བ་དང་། ཉེ་བར་གནོད་པར་བྱ་བ་དང་། ཚར་གཅད་པར་བྱ་བའི་ཕྱིར་འོང་བར་ཡང་མི་འགྱུར་རོ། །དེ་ལ་དད་ཅིང་དང་བ་རྣམས་དབབ་པའི་ཚིག་གང་ཞེ་ན། །གུ་རུ་དྷ་རེ། ས་ཙ་ལེ། སྭ་སྟི་ནིར་ཧ་ར་ཀོ་བི་དེ། ནིར་ཛ་ན་པྲ་ཏི་བི་ལོ་ཡི་ཏེ། ཙིཏ་ཏ་སན་ཙ་ན་ནི། ཙིཏ་ཏ་སམ་པྲ་སཱ་དྷ་ནི། མ་ན་སྱ་ཧ་རི་ཤ་ཀ་ར་ཎི། བཱིད་ཛྙཱ་ན་སྱ་ཨ་ནུ་སྭ་ཏི། ཨ་ནུ་དྷརྨོ་ནོ་ཨུ་པེ་ཀྵ་ཎེ། ཧེ་ཏུ་སན་ད་རི་ཤ་ནེ། ཏཐཱ་ཨ་ཀྵ་ར་པ་ད། ནིར་དེ་ཤ་ཤོ་ཐིགཡ་ཐཱ་གཱ་ཐཱ། ཡ་ཐཱ་ནུ་ཀུ་ལ་ས་མ་ཨུ་པ་མ་ནིར་དེ་ཤ་ཤོ་དྷི་ཏེ། ད་ཏྲ་ཀཱ་ལེ་ན་ཙ་ཀར་ཏ་བ་བྱ། སམ་པྲ་སཱ་ད་ན་ཨརྠ་ཏོ་ས་མ། ཡ་ཐཱ་ཨ་ནུ་ཀུ་ལན་ཙ་མུ་ཁ། ཨུ་པསཾ་སྭ་ར་ཡོ་ནི་ཤ། ཨཱ་ཀཱ་ཤ་སྱ་བི་ཤུད་དྷི། ཡ་ཡ་ཐཱ་ཤོ་དྷི་ཏི་ལཀྵ་ཎ། ལཀྵ་ཎི། ལཀྵ་ཎ་བ་ཏི། ལཀྵ་ཎ་བི་ཤོ་དྷ་ནི། པྲ་ཏི་བེདྷ་ཨརྠ་སན་དྷ་ད་རི་ཤ་ནི། ཀུ་ཤ་ལ་བྷོ་ཐ་སར་བེ་ཧ་པྲ་བི་ཙ་ཡི། ས་མོ་ས་ར་ནི། ཀྲིཏ་ཨ་ནུ་ར་ཀྵི་ས་ཏྱ་ཨརྠེ། ས་ཏྱ་ནཱཾ་སུ་བི་ཤོ་དྷི་ཏེ། བཀོད་པ་མཐའ་ཡས་འདི་དག་ནི་དང་བ་རྣམས་དབབ་པ་ཉེ་བར་སྒྲུབ་པའི་ཚིག་ཡིན་ཏེ། འདི་དག་གིས་ཆོས་
སྨྲ་བ་དེ་དག་ལ་འདི་ལས་ཇི་ལྟར་བསྟན་པ་བཞིན། དགེ་བའི་རྩ་བ་དོན་དང་ལྡན་པ་ཉེ་བར་སྒྲུབ་པར་བྱེད་པར་འགྱུར་རོ། །དེ་ལ་མ་དད་པ་རྣམས་ཚར་གཅད་པའི་ཚིག་གང་ཞེ་ན། ཀྵ་མེ། ཀྵ་མ་བ་ཏི། མཻ་ཏྲེ་པ་རི། ཀར་མ་པྲ་ཏྱུ་པ་སྠཱ་ནེ། ཀ་རུ་ཎ་པྲ་ཏི་ལ་བྷ་ཧི་ཏ་ཨ་ནུ་ཀམ་པ་སཉྫ་ན་ནི།སཾ་གྲ་ཧ་བ་སྟུ་ཧི་ཏ་བ་སྟུ། སན་ད་རི་ཥ་ཎི་པ་རི་བརྫ་ནི། ཡ་བརྫ་ནི། པཱ་བ་མི་ཏྲ་སེ་བ་ནི། ན་ཏ་ཏྲ་དེ་ཤེ་སྠཱ་ཏ་བ་བྱ་ཡ་ཏྲ་བི་གྲ་ཧ་བརྟི་ཏོ། སརྦ་གྲན་ཐ་པྲ་ད་ལ་ཏི། ཨུ་པ་ཙཱ་ར་སན་ད་རི་ཤ་ཎི། ནི་ཤ་ཡ་བི་ཤོ་དྷ་ནི། ཨཱད་མ་ཨ་ནུ་ག་མེ། པ་ར་ཨུ་པ་སཾ་ཧ་ར་ཎི། ནིར་མི་ན་ཎི།ཨ་ནུ་ར་ཀྵ་པྲ་ཏྱུ་པ་སྠཱ་ནེ། བཀོད་པ་མཐའ་ཡས་འདི་དག་ནི་སེམས་ཅན་མ་དད་ཅིང་མ་དང་བ་རྣམས་ཚར་གཅད་པའི་ཚིག་ཡིན་ཏེ། དེས་ན་དེ་དག་ཉེས་པ་མཐོང་བར་འགྱུར་རོ། །བཀོད་པ་མཐའ་ཡས་ཆོས་དགེ་བ་ལ་ཡང་ཕྱིར་རྒོལ་བ་མང་བ་ཡིན་ན། ཆོས་འདུལ་བ་བླ་ན་མེད་པ། ཉེས་པ་ཐམས་ཅད་འཇོམས་པར་འགྱུར་བ། རྩོད་པའི་མདུད་པ་ཐམས་ཅད་སྤོང་བར་འགྱུར་བ་འདི་ལ་ལྟ་སྨོས་ཀྱང་ཅི་དགོས། བཀོད་པ་མཐའ་ཡས་སེམས་ཅན་མ་དད་ཅིང་མ་དང་བ་དེ་དག་ཀླན་ཀ་བཙལ་བའི་བསམ་པ་དང་། གནོད་སེམས་ཀྱི་བསམ་པ་མང་པོ་དང་། རྣམ་པར་ཐོ་བཙམ་པའི་བསམ་པས་ངའི་དྲུང་དུ་ཉེ་བར་འོང་བ་ལ་ལྟོས། དེ་བཞིན་གཤེགས་པ་ནི་དེ་དག་གི་སེམས་ཀྱི་སྤྱོད་པ་དང་། གཡོ་བ་མཁྱེན་ནས་ཅི་ནས་ཀྱང་དགེ་བའི་རྩ་བ་དང་ལྡན་ཞིང་རྣམ་པར་རྟོག་པ་སྤོང་བ་དང་། དགེ་བའི་རྩ་བའི་རྒྱུ་ཡང་ཡོངས་སུ་འཛིན་པར་འགྱུར་བ་དེ་ལྟ་བུའི་ཆོས་ཀྱི་སྒོ་དག་གིས་ཡང་དག་པར་སྐུལ་བར་མཛད་དོ། །བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པས་ནི་སེམས་ཅན་ཐམས་ཅད་ཀྱི་སེམས་ཀྱི་སྤྱོད་པ་ཡང་རབ་ཏུ་མཁྱེན་ལ། སེམས་ཡང་དག་པར་བསྡུ་བ་ཡང་རབ་ཏུ་མཁྱེན་ཏོ། །བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་ནི་སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་བཅྭ་བརྒྱད་ལ་གནས་པ་ཡིན་ནོ། །བཅྭ་བརྒྱད་པོ་གང་དག་ལ་གནས་ཤེ་ན། བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་གང་གི་ནུབ་མོ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ནས་ལེན་པ་མ་ལུས་པའི་མྱ་ངན་ལས་འདས་པའི་དབྱིངས་སུ་ཡོངས་སུ་མྱ་ངན་ལས་འདའ་བར་འགྱུར་བ་དེའི་བར་དུ་དེ་བཞིན་གཤེགས་པ་ལ་འཁྲུལ་པ་མི་མངའོ། །ཅ་ཅོ་མི་མངའོ། །བསྙེལ་བ་མི་མངའོ། །སོ་སོར་མ་བརྟགས་པའི་འགོག་པ་མི་མངའོ། །འདུ་ཤེས་ཐ་དད་པ་མི་མངའོ། །སེམས་མཉམ་པར་མ་བཞག་པ་མི་མངའོ། །བརྩོན་འགྲུས་ཉམས་པ་མི་མངའོ། །དྲན་པ་ཉམས་པ་མི་མངའོ། །འདུན་པ་ཉམས་པ་མི་མངའོ། །ཏིང་ངེ་འཛིན་ཉམས་པ་མི་མངའོ། །ཤེས་རབ་ཉམས་པ་མི་མངའོ། །རྣམ་པར་གྲོལ་བ་ཉམས་པ་མི་མངའོ། །རྣམ་པར་གྲོལ་བའི་ཡེ་ཤེས་མཐོང་བ་ཉམས་པ་མི་མངའོ། །དེ་བཞིན་གཤེགས་པའི་སྐུའི་ཕྲིན་ལས་ཐམས་ཅད་ཡེ་ཤེས་སྔོན་དུ་འགྲོ་ཞིང་ཡེ་ཤེས་ཀྱི་རྗེས་སུ་འབྲང་བ་ཡིན། དེ་བཞིན་གཤེགས་པའི་གསུང་གི་ཕྲིན་ལས་ཐམས་ཅད་ཡེ་ཤེས་སྔོན་དུ་འགྲོ་ཞིང་ཡེ་ཤེས་ཀྱི་རྗེས་སུ་འབྲང་བ་ཡིན། དེ་བཞིན་གཤེགས་པའི་ཐུགས་ཀྱི་ཕྲིན་ལས་ཐམས་ཅད་ཡེ་ཤེས་སྔོན་དུ་འགྲོ་ཞིང་ཡེ་ཤེས་ཀྱི་རྗེས་སུ་འབྲང་བ་ཡིན། དེ་བཞིན་གཤེགས་པ་ནི་འདས་པའི་དུས་ལ་མ་ཆགས་མ་ཐོགས་པའི་ཡེ་ཤེས་མཐོང་བ་ལ་འཇུག་པ་ཡིན། དེ་བཞིན་གཤེགས་པ་ནི་མ་འོངས་པའི་དུས་ལ་མ་ཆགས་
མ་ཐོགས་པའི་ཡེ་ཤེས་མཐོང་བ་ལ་འཇུག་པ་ཡིན། དེ་བཞིན་གཤེགས་པ་ནི་ད་ལྟར་བྱུང་བའི་དུས་ལ་མ་ཆགས་མ་ཐོགས་པའི་ཡེ་ཤེས་མཐོང་བ་ལ་འཇུག་པ་ཡིན་ཏེ། དེ་བཞིན་གཤེགས་པ་ནི་སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་བཅྭ་བརྒྱད་པོ་དེ་དག་དང་ལྡན་པ་ཡིན་ནོ། །བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་ཡེ་ཤེས་མཐོང་བའི་སྟོབས་ཚད་མེད་པ་དང་ལྡན་པས་སེམས་ཅན་མ་དང་བ་རྣམས་དང་བར་བྱ་བ་དང་། སེམས་ཅན་དང་བ་རྣམས་ཆོས་འདུལ་བ་འདི་ལ་ཡེ་ཤེས་རྣམ་པར་དག་པ་ཐོབ་པར་བྱ་བའི་ཕྱིར་གཟུངས་ཀྱི་སྒོ་རྣམ་པར་སྦྱོང་བའི་ཆོས་རིམ་པར་ཕྱེ་བ་སྟོན་པ་རྗོད་པར་མཛད་དོ། །བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་བཞེད་ན་གཟུངས་ཀྱི་སྒོའི་ཆོས་རིམ་པར་ཕྱེ་བ་ཚིག་ཏུ་བརྗོད་པ་ཚད་མེད་པ་སྟོན་པར་མཛད་དེ། དེ་བཞིན་གཤེགས་པས་ཚད་མེད་པ་བསྟན་ཀྱང་གཟུངས་ཀྱི་སྒོ་བསྟན་པའི་ཆོས་ཐེ་ཚོམ་ཐམས་ཅད་གཅོད་པར་བྱེད་པ་འདིའི་བརྒྱའི་ཆ་ནས་བྱེ་བ་ཕྲག་བརྒྱ་སྟོང་གི་ཆ་དང་། གྲངས་དང་། དཔེ་དང་། ཆ་ཙམ་ཡང་ཉེ་བར་བསྟན་པར་མི་འགྱུར་རོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཆོས་ཀྱི་སྒོ་འདི་ནི་སྒོ་ཚད་མེད་པ་ཡིན། སྒོ་བསམ་གྱིས་མི་ཁྱབ་པ་ཡིན་ཏེ། སྒོ་འདིས་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཐོབ་པར་འགྱུར་རོ། །བཀོད་པ་མཐའ་ཡས་དེའི་ཕྱིར་དེ་བཞིན་གཤེགས་པས་ཅི་ནས་ཀྱང་ཁྱེད་ཀྱིས་ཆོས་ཀྱི་སྒོ་ཚད་མེད་ཅིང་མཐའ་ཡས་པ་འདི་ཀུན་ཤེས་པར་འགྱུར་ཞིང་གཟུངས་ཐོབ་པར་བྱ་བའི་ཕྱིར་ཆོས་ཀྱི་སྒོ་འདི་རྣམ་གྲངས་དུ་མས་ཁྱེད་ལ་སྟོན་པར་མཛད་དོ། །ཨཱ་རྱེ། ཨཱརྱ་བ་ཏི། ཨཱརྱ་ཨ་ནུ་ག་ཏེ། ནི་ཐཱ་ནེ། ནི་ཐཱ་ན་བ་ཏི། བར་ཏོ་བར་མེ། བར་ཏོ་བ་ཙྪེ་ད་ན་ཀ་ར་ཎི། ཨཱ་ཀཱ་ཤ་བི་ཤོ་དྷ་ནི། ཨ་ནུ་ཀྵེ་པ་ནི། ཨ་བཱི་ག་ཏ་པར་མེ། ཨ་ནུ་པ་ག་ཏི་ནི། ནི་རྦཱ་ཎ་པ་ཐ་བི་ཤོ་དྷ་ནི། བྱ་བ་ནི་ཀྵིཔ་ཏི། ཨ་ནུད་པཱ་ད་བི་རོ་དྷ་པར་མེ། སརྦ་ཛྙཱ་ནི་རྦཱ་ཎ་སན་ད་ར་ཤ་ནི། བཀོད་པ་མཐའ་ཡས་འདི་དག་ནི་གཟུངས་ཀྱི་སྒོ་མངོན་པར་སྒྲུབ་པ་སྟོན་པའི་ཆོས་རིམ་པར་ཕྱེ་བ་འདིར་གཟུངས་ཀྱི་ཕྱག་རྒྱ་རྣམ་པར་སྦྱོང་བའི་ཚིག་གི་རྣམ་གྲངས་བསྟན་པ་ཡིན་ཏེ། ཚིག་འདི་དག་ཀུན་བཟུང་བས་བྱང་ཆུབ་སེམས་དཔའ་ཚེགས་ཆུང་དུས་ཚིག་རབ་ཏུ་དབྱེ་བ་ཤེས་པ་མཁས་པ་ཡང་རྟོགས་པར་འགྱུར། སྙིང་རྗེ་ཆེན་པོ་དང་ཡང་ཉེ་བར་འགྱུར། དོན་གྱི་རྗེས་སུ་སོང་བའི་བློས་ཆོས་ཐམས་ཅད་ངེས་པར་འབྱེད་པའི་ཤེས་རབ་ཀྱང་རྟོགས་པར་འགྱུར་རོ། །བཀོད་པ་མཐའ་ཡས་ཚིག་འདི་དག་ནི་སྨན་ཆེན་པོ་ཡིན་ཏེ། མ་རིག་པ་དང་མི་ཤེས་པའི་ནད་ཀྱི་མུན་པ་མུན་ནག་སྨག་ཏུ་གྱུར་པ་རྣམས་ཉེ་བར་ཞི་བར་བྱེད་ཅིང་རིག་པ་དང་འཐུན་པའི་ཆོས་རྣམས་ཡོངས་སུ་རྫོགས་པར་བྱེད་པར་འགྱུར་བ་ཡིན། སྔོན་གྱི་གནས་རྗེས་སུ་དྲན་པ་མངོན་དུ་བྱ་བ་དང་འཐུན་པའི་ཆོས་ཤེས་པ་ལ་མཁས་པ་ཡོངས་སུ་རྫོགས་པར་བྱེད་པ་ཡིན། ལྷའི་མིག་མངོན་པར་སྒྲུབ་པ་རིག་པ་དང་འཐུན་པའི་ཆོས་ཤེས་པ་ཐོབ་པར་བྱེད་པ་ཡིན། ཟག་པ་ཟད་པ་ཤེས་པ་མངོན་དུ་བྱ་བ་རིག་པ་དང་འཐུན་པ་ཕུང་པོ་ཐམས་ཅད་ངེས་པར་སྤོང་བར་བྱེད་པ་ཡིན། བསླབ་པ་ཐམས་ཅད་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དང་། ཡེ་ཤེས་མཐོང་བ་བླ་ན་མེད་པ་དང་། ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་མཐོང་བ་དང་། ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཀྱིས་ཐོབ་པར་བྱེད་པ་ཡང་ཡིན་ནོ། །བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པའི་ཆོས་བསྟན་པ་རྒྱ་ཆེན་པོ་ཇི་ཙམ་དུ་ཐབས་ལ་མཁས་པ་ཐམས་ཅད་ཡོངས་སུ་རྫོགས་པར་འགྱུར་བ་ལ་ལྟོས། བཀོད་པ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་ནི་ཡེ་ཤེས་ཆེན་པོ་དང་ལྡན་ཞིང་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་དང་།
 སྟོབས་དང་མི་འཇིགས་པ་ལ་ཤིན་ཏུ་གནས་ཏེ། ཆོས་རིན་པོ་ཆེའི་མཛོད་བླ་ན་མེད་པ། བསྐལ་པ་བྱེ་བ་འབུམ་ཕྲག་དུ་མར་དགེ་བའི་རྩ་བས་ཡང་དག་པར་བསྒྲུབས་པ་འདི་ངེས་པར་བསྟན་ཅིང་རབ་ཏུ་ཕྱེའོ། །དེའི་ཕྱིར་དེ་བཞིན་གཤེགས་པས་ཁྱོད་ལ་སོགས་པ་སངས་རྒྱས་ཀྱི་ཆོས་ལ་ཡོངས་སུ་སྨིན་པར་བྱ་བ་དང་། ཆོས་ཀྱི་ཚུལ་མངོན་པར་བསྒྲུབ་པ་ལ་མཁས་པར་བྱ་བའི་ཕྱིར་ཆོས་ཀྱི་སྒོ་འདི་ལ་རབ་ཏུ་གནས་ཏེ། གཟུངས་ཀྱི་སྒོ་རྣམ་པར་སྦྱོང་བའི་ཆོས་རིམ་པར་ཕྱེ་བ་བསྟན་པ་འདི་བྱིན་གྱིས་རློབ་པར་མཛད་དོ། །བཀོད་པ་མཐའ་ཡས་རིགས་ཀྱི་བུ་འམ་རིགས་ཀྱི་བུ་མོ་བྱང་ཆུབ་དོན་དུ་གཉེར་བ་ངའི་རྗེས་སུ་སློབ་པར་འདོད་པ། ངའི་ཆོས་ཀྱི་ཚུལ་རྗེས་སུ་སློབ་པར་འདོད་པ། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་ཆོས་ཀྱི་གཏེར་ཚད་མེད་པ་ཀུན་གཟུང་བར་འདོད་པ་རྣམས་ཀྱིས་ཆོས་བསྟན་པ་འདི་ལ་མངོན་པར་བརྩོན་པར་བྱའོ། །བཀོད་པ་མཐའ་ཡས་འདི་ལ་འདུན་པ་དང་། བརྩོན་འགྲུས་དང་། བག་ཡོད་པའི་ཚུལ་ཅན་ཁམས་གསུམ་པ་ཐམས་ཅད་ལ་མི་ལྟ་བ། ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཀྱི་སེམས་ཡིད་ལ་བྱེད་པས་ལེགས་པར་བྱིན་གྱིས་བརླབས་པས་གཟུགས་ལ་དམིགས་པ་རྣམ་པར་དག་པ་ལ་མཁས་པར་བརྩོན་པ། ཚོར་བ་དང་། འདུ་ཤེས་དང་། འདུ་བྱེད་དང་། རྣམ་པར་ཤེས་པ་ལ་དམིགས་པ་རྣམ་པར་དག་པ་ལ་མཁས་པར་བརྩོན་པ། བདེན་པའི་ཚིག་རྣམ་པར་བསྒྲུབ་པ་རྣམ་པར་དག་པ་ལ་མཁས་པར་བརྩོན་པ་རྣམས་ཀྱིས་ཆོས་ཐམས་ཅད་ལ་རྣམ་པར་དག་པའི་ཡེ་ཤེས་འདོད་པར་བྱའོ། །ནང་རྣམ་པར་དག་པའི་ཕྱིར་ཆོས་ཐམས་ཅད་རྣམ་པར་དག་པའོ། །ནང་དབེན་པའི་ཕྱིར་ཆོས་ཐམས་ཅད་དབེན་པའོ། །ནང་ཉེ་བར་ཞི་བའི་ཕྱིར་ཆོས་ཐམས་ཅད་ཉེ་བར་ཞི་བའོ། །ནང་ཉེ་བར་ལེན་པ་མེད་པའི་ཕྱིར་ཆོས་ཐམས་ཅད་ཉེ་བར་ལེན་པ་མེད་པའོ། །ནང་སྤོང་བའི་ཕྱིར་ཆོས་ཐམས་ཅད་སྤོང་བའོ། །ནང་འགོག་པའི་ཕྱིར་ཆོས་ཐམས་ཅད་འགོག་པའོ། །ནང་འདུ་བྱེད་པ་མེད་པས་ཆོས་ཐམས་ཅད་འདུ་བྱེད་པ་མེད་པའོ། །བཀོད་པ་མཐའ་ཡས་བྱང་ཆུབ་སེམས་དཔའི་སྒོ་འདི་དག་ནི་ནང་རྣམ་པར་དག་པར་བྱེད་པར་འགྱུར་བ་ཡིན་ནོ། །ཕྱི་རོལ་རང་བཞིན་གྱིས་རྣམ་པར་མི་རྟོག་པའི་ཕྱིར་གང་དུ་ཡང་རློམ་པར་མི་བྱེད་པས། གཟུངས་ཀྱི་སྒོ་ཡོངས་སུ་དག་པ་ཡང་ཀུན་འཛིན་པར་འགྱུར། ང་རྒྱལ་མེད་པ་ཡང་འཐོབ་པར་འགྱུར། རྗེས་སུ་ཆགས་པ་དང་། །ཁོང་ཁྲོ་བ་ཡང་སྤོང་བར་འགྱུར། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་ཀྱང་བསྔགས་པར་འགྱུར། སེམས་ཅན་ཐམས་ཅད་ཀྱི་མཆོག་ཏུ་ཡང་འགྱུར། དབང་བླ་ན་མེད་པ་ཡང་བྱེད་པར་འགྱུར། ཡེ་ཤེས་ཆགས་པ་མེད་པ་བསྟན་པ་ལ་ཡང་དགའ་ཞིང་སྤོབས་པ་འཐོབ་པར་འགྱུར། སྔོན་གྱི་མཐའ་དང་། ཕྱི་མའི་མཐའ་རྣམ་པར་དག་པ་ཤེས་པ་ཡང་ཀུན་འཛིན་པར་འགྱུར། དབུས་ངེས་པར་ཞི་བ་ཡང་རྗེས་སུ་དྲན་པར་འགྱུར། མི་སྐྱེ་བའི་ཆོས་ལ་བཟོད་པ་ཐོབ་པ་ལ་བརྟེན་ནས་སྨོན་ལམ་གྱི་ཁྱད་པར་ཡང་རྟོགས་པར་འགྱུར། སྨོན་ལམ་ཡོངས་སུ་དག་པ་ཡང་ཐོབ་པར་འགྱུར། ཆོས་ཐམས་ཅད་ཤེས་པ་ལ་མཁས་པ་ཐུན་མོང་མ་ཡིན་པ་ཡང་ཀུན་འཛིན་པར་འགྱུར། ཚིག་བཙུན་ཞིང་ཆོས་ཀྱི་ཆར་ཆེན་པོ་ཡང་འབེབས་པར་འགྱུར་རོ། །བཀོད་པ་མཐའ་ཡས་བྱང་ཆུབ་སེམས་དཔའི་སྨོན་ལམ་གྱི་ཁྱད་པར་ཐོབ་ནས་མི་སྐྱེ་བའི་ཆོས་ལ་བཟོད་པ་ཐོབ་པར་འགྱུར་ཞིང་ཆོས་ཐམས་ཅད་རྣམ་པར་དག་པ་ཡང་ཀུན་ཏུ་འཛིན་པར་འགྱུར་ལ་འདི་ལྟར་ཆོས་ཐམས་ཅད་ནི་སྐྱེ་བ་མེད་ཅིང་འཇིག་པ་མེད་དོ་ཞེས་བྱ་བའི་ཆོས་ཤེས་པ་ཡང་མངོན་པར་སྒྲུབ་པར་འགྱུར། འདི་ལྟར་ཆོས་འདི་དག་ཐམས་ཅད་ནི་མ་བྱུང་བ་ལས་བྱུང་བ། བྱུང་ནས་ཀྱང་ཡོངས་སུ་གྲུབ་པར་མི་འགྱུར་གྱི། ཡོངས་སུ་མ་གྲུབ་པ་ཁོ་ནར་འབྲལ་བར་འགྱུར་བས་འབྲལ་བའི་ཆོས་སུ་ཡང་དག་པར་འདུ་ཞིང་འབྲལ་བའི་སྒོ་ཅན་ནོ་ཞེས་བྱ་བར་ཆོས་ཤེས་པ་ཡང་ཐོབ་པར་འགྱུར་རོ། །བྲལ་བ་ནི་མི་སྐྱེ་བའི་མཚན་ཉིད་དུ་ཡང་དག་པར་འདུའོ། །ཆོས་ཐམས་ཅད་མི་སྐྱེ་བའི་མཚན་ཉིད་གང་ཡིན་པ་དེ་ནི་འགོག་པའི་མཚན་ཉིད་དང་། འདོད་ཆགས་དང་བྲལ་བའི་མཚན་ཉིད་དེ། དེ་ལྟར་མངོན་པར་སྒྲུབ་པ་ནི་མངོན་པར་ཞེན་པར་ཡང་མི་བྱེད་སྤྲོས་པར་ཡང་མི་བྱེད་དོ། །བཀོད་པ་མཐའ་ཡས་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཆོས་མི་སྐྱེ་བ་ཤེས་པའི་སྒོ་ལ་མཁས་པ་ཡོངས་སུ་རྟོགས་པའི་གཟུངས་ཀྱི་སྒོར་འཇུག་པ་ཡིན་ཏེ། དེས་ན་མྱུར་དུ་མི་སྐྱེ་བའི་ཆོས་ལ་བཟོད་པ་ཡང་ཐོབ་པར་འགྱུར། སྤོབས་པ་ཕུན་སུམ་ཚོགས་པ་ཡང་ཀུན་ནས་འདྲེན་པར་ཡང་འགྱུར་རོ། །དེ་ནས་བཅོམ་
ལྡན་འདས་ཀྱིས་ཆོས་ཀྱི་སྒོ་བསྒྲུབ་པ་འདི་ཡང་དག་པར་བརྗོད་པའི་ཕྱིར་ཚིགས་སུ་བཅད་པ་འདི་དག་གསུངས་སོ། །མ་བྱུང་བ་ལས་ཆོས་འབྱུང་སྟེ། །བྱུང་ནས་སོ་སོར་འབྲལ་བར་འགྱུར། །བྱུང་ནས་འབྲལ་བའི་དོན་ཡིན་ན། །འདི་ལ་གང་ཞིག་འཛིན་པར་འགྱུར། །ཆོས་ཀུན་དངོས་པོ་མེད་པ་སྟེ། །ཉོན་མོངས་མེད་ཅིང་ལེན་པ་མེད། །གང་ལ་ཆོས་ཀྱང་མི་རྙེད་པ། །དེ་དག་ཇི་ལྟར་ཀུན་ཏུ་འཛིན། །ཆོས་ཀུན་ངོ་བོ་ཉིད་ཀྱིས་ནི། །ཐོབ་པ་མེད་ཅེས་རློམ་པ་གང་། །དེ་དག་མཚན་མ་སྤྱོད་པ་སྟེ། །གཟུངས་འདི་ཐོབ་པར་མི་འགྱུར་རོ། །དེ་ཡི་ཕྱིར་ན་ཆོས་རྣམས་ཀུན། །ནམ་མཁའ་ལྟ་བུར་རབ་ཏུ་གསལ། །ནམ་མཁའ་དག་ནི་རབ་གསལ་བ། །དེ་དག་གཉི་ག་ཡོད་མ་ཡིན། །ཆོས་འདི་ཐམས་ཅད་ཡོད་མིན་ཞིང་། །མེད་པ་དང་ཡང་བྲལ་བ་ཡིན། །གང་གིས་དེ་ལྟར་ཆོས་ཤེས་པ། །དེ་དག་ལ་ནི་གཟུངས་སྐྱེའོ། །གང་ཞིག་མཐའ་མེད་རྗེས་རྟོགས་ཤིང་། །དབུས་སུའང་རློམ་པར་མི་བྱེད་པ། །དེ་དག་རློམ་སེམས་བྲལ་བ་སྟེ། །ཆོས་རྣམས་ཐམས་ཅད་མེད་པར་བཤད། །གང་ལ་སྙིང་པོ་མི་རྙེད་པ། །དེ་ལ་སྙིང་པོ་ཡོད་མིན་ན། །ཆོས་རྣམས་ཐམས་ཅད་ཇི་ལྟ་བུར། །ཡང་དག་ཚུལ་གྱིས་ཀུན་ཏུ་འཛིན། །དེ་ལྟར་ཆོས་རྣམས་རང་བཞིན་གྱིས། །ཉོན་མོངས་མེད་པར་ཤེས་པ་གང་། །དེ་དག་རྣམས་ལ་མདོར་ན་འདིར། །རྣམ་པར་དག་པའི་གཟུངས་སུ་འགྱུར། །གང་ཞིག་ཆོས་ཀུན་མཁའ་འདྲ་ཞིང་། །རྟག་ཏུ་བར་སྣང་ལྟ་བུར་ཡང་། །ཤེས་རབ་ཀྱིས་ནི་ངེས་རྟོགས་པ། །དེ་ལ་གཟུངས་ཀུན་སྐྱེ་བར་འགྱུར། །ཆོས་རྣམས་ཐམས་ཅད་མ་སྐྱེས་ཤིང་། །མ་བྱུང་ཉོན་མོངས་མེད་པ་ཡིན། །ཉོན་མོངས་མེད་ཅིང་ལེན་མེད་པ། །འདིར་ནི་ཇི་ལྟར་ཀུན་ཏུ་འཛིན། །ཆོས་ཀུན་ངོ་བོ་ཉིད་ཀྱིས་ནི། །སྤྲོས་པ་མེད་ཅིང་མཚན་ཉིད་མེད། །ཆོས་ཀུན་མཚན་ཉིད་བྲལ་བས་ན། །ཉོན་མོངས་མེད་པར་བཤད་པ་ཡིན། །གང་གིས་དེ་ལྟར་ཆོས་ཤེས་ཤིང་། །གང་དུའང་རློམ་པར་མི་བྱེད་པ། །དེ་ཡིས་ཆོས་ཀུན་ཚུལ་བཞིན་དུ། །ཀུན་ཏུ་འཛིན་པར་འགྱུར་བ་ཡིན། །གང་གིས་ཆོས་ཀུན་རང་བཞིན་གྱིས། །མེད་དེ་ཡོད་པ་མ་ཡིན་དང་། །ཡོད་པ་མ་ཡིན་དོན་རྟོགས་པ། །དེ་ལ་གཟུངས་ནི་ཉེ་བར་སྐྱེ། །དཔའ་བོ་གང་ཞིག་དེ་ལྟ་བུར། །སོ་སོར་རྟོགས་པ་དེ་ལ་ནི། །ཆོས་རྣམས་ཀུན་གྱིས་མི་གོས་ཤིང་། །སྟོང་པར་མི་རློམ་ཆོས་ཀུན་འཛིན། །གང་ཞིག་མི་རྟག་ཉིད་དོན་དང་། །མེད་དོན་སྡུག་བསྔལ་ཉིད་དོན་དུ། །འདོད་ཆགས་བྲལ་བར་རྣམ་ཤེས་པ། །དེ་ལ་ཤེས་རབ་ཡེ་ཤེས་འཕེལ། །བློ་གྲོས་བརྟན་པ་གང་ཞིག་གིས། །ལེན་མེད་མྱ་ངན་འདས་དོན་དུ། །ཆོས་ཀྱིས་ཚུལ་བཞིན་འདུལ་བྱེད་ཅིང་། །མི་རྟོག་རྣམ་པར་མི་རྟོག་པ། །དེས་ནི་སྙིང་པོ་མེད་པའི་ཆོས། །ཉོན་མོངས་མེད་ཅིང་ལེན་པ་མེད། །དབེན་ཞིང་སྟོང་ལ་བལྟ་དཀའ་བ། །ཐམས་ཅད་ཀུན་ཏུ་འཛིན་པར་འགྱུར། །གང་གིས་ཆོས་སུ་བརྗོད་ཤེས་ཤིང་། །བརྗོད་པ་ལ་ཡང་མི་རློམ་ལ། །ཀུན་ཏུ་འཛིན་པས་ལེན་མེད་པ། །རློམ་སེམས་མེད་པའི་ཚུལ་འདི་ཡིན། །གང་གིས་མཚན་མ་རྣམ་བཤིག་ནས། །མཚན་མ་མེད་པ་རབ་ཤེས་ཤིང་། །ཆོས་རྣམས་ཀུན་གྱི་སྤོང་བ་ཡི། །འདུ་ཤེས་འཕེལ་བར་མི་འགྱུར་བ། །དེས་ནི་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས། །བཤད་པའི་དོན་འདི་རྣམ་ཤེས་ཤིང་། །དགོངས་ཏེ་བསྟན་ལ་མཁས་པ་ཡང་། །མཉམ་པར་རྗེས་སུ་རྟོགས་པར་འགྱུར། །གང་གིས་ཆོས་ཀུན་ཚད་མེད་པར། །ཚུལ་བཞིན་དུ་ནི་ངེས་རྟོགས་པ། །དེ་ཡིས་ཚུལ་འདི་ཚད་དག་དང་། །བྲལ་ཞིང་མཉམ་པར་རྟོགས་པར་འགྱུར། །གང་གིས་ཆོས་རྣམས་མིང་མེད་ཅིང་། །མཚན་མ་མེད་པར་ཡོངས་རྟོགས་པ། །དེ་འདིར་བཟོད་པ་འཕེལ་འགྱུར་ཞིང་། །དོན་འདི་རྣམ་པར་ཤེས་པར་འགྱུར། །ཚུལ་བཞིན་དུ་ནི་སྨོན་ལམ་གྱི། །ཁྱད་པར་ཡོངས་སུ་རྣམ་རྟོག་ཅིང་། །སྨོན་པ་ཡི་ནི་ངོ་བོ་དང་། །འགལ་བར་
འགྱུར་བ་མ་ཡིན་ནོ། །བརྟན་པ་དེ་ནི་དོན་གྱི་ལུགས། །འདི་ལ་ཚུལ་བཞིན་ཡོངས་རྟོག་ཀྱང་། །དེ་ལ་ཆོས་རྣམས་ཐམས་ཅད་ཀྱིས། །ཚུལ་བཞིན་དོགས་པར་མི་འགྱུར་རོ། །གང་གི་ཤེས་རབ་ཆོས་ཀུན་གྱི། །རྣམ་པ་ངེས་པར་རྟོགས་འགྱུར་ཞིང་། །གང་ཞིག་རྣམ་པ་མེད་པ་ལ། །འཇུག་པ་ངེས་པར་རྣམ་ཤེས་པ། །དེས་ནི་བསྟན་པ་འདི་ཡི་ཚུལ། །བྱེ་བྲག་ཤེས་ཤིང་རབ་ཏུ་གནས། །སངས་རྒྱས་ཀྱི་ནི་ཆོས་རྣམས་དང་། །མི་འཇིགས་མྱུར་དུ་ཐོབ་པར་འགྱུར། །གང་ཞིག་ཆོས་ལ་སྤྲོས་མེད་ཅིང་། །མཉམ་པར་ཡང་ནི་མི་རྟོག་པ། །དེ་ཡིས་ཆོས་ཀུན་སྦྱོར་ཤེས་ནས། །འདོད་ཆགས་བྲལ་ལ་རྨོངས་མི་འགྱུར། །གང་དག་འགོག་དང་ཞི་བའི་དོན། །ཕུང་པོ་ཟད་པར་མ་རློམ་པ། །དེ་དག་གིས་ནི་ཚུལ་བཞིན་དུ། །མཉམ་པའི་ཆོས་འདི་ཐོབ་པར་འགྱུར། །བྱམས་པ་དང་ནི་སྙིང་རྗེ་དག །སྒོམ་པར་བྱེད་ལ་རབ་གནས་ཤིང་། །སེམས་ཅན་ཀུན་ལ་ཕན་བརྩོན་གང་། །དེ་དག་བླ་མེད་རྟོགས་པར་འགྱུར། །སེམས་ཅན་འདུ་ཤེས་རྣམ་གྲོལ་ཞིང་། །སེམས་ཅན་མེད་པའི་ཆོས་ཤེས་གང་། །དེ་ཡིས་ཆོས་རྣམས་སྤྲོས་མེད་པའི། །ཚུལ་བཞིན་དོན་ལ་སྤྲོས་མི་བྱེད། །གང་གིས་ཆོས་འདི་ཐོས་གྱུར་ནས། །དད་པ་མྱུར་དུ་ཐོབ་འགྱུར་བ། །དེས་ནི་བྱམས་པ་རྐང་གཉིས་མཆོག །རྫོགས་པའི་སངས་རྒྱས་མཐོང་བར་འགྱུར། །གང་དག་འདི་དང་ཕྲད་གྱུར་ཅིང་། །བཟང་པོས་ཆོས་འདི་ཐོས་ནས་ནི། །དགའ་བར་བྱེད་པ་དེ་དག་གིས། །ང་ཡང་ཀུན་ཏུ་མཉེས་པར་བྱས། །གང་དག་དེ་བཞིན་གཤེགས་པ་ལ། །དད་པ་མི་ཕྱེད་ཐོབ་འགྱུར་བའི། །བཟང་པོ་དེ་དག་ཆོས་འདི་རྣམས། །ཐོས་ནས་དགའ་བ་བྱེད་པར་འགྱུར། །གང་དག་བསྐལ་པ་བཟང་པོ་ཡི། །རྒྱལ་བ་བལྟ་བར་འདོད་དེ་དག །ཚུལ་འདི་ལ་ནི་ལེགས་བསླབས་ན། །རྒྱལ་བ་དེ་དག་མཉེས་པར་འགྱུར། །ཚེ་དཔག་མེད་དང་གཟི་བརྗིད་ཆེ། །མི་འཁྲུགས་པ་དང་གྲགས་ཆེན་དག །བལྟ་བར་འདོད་པ་གང་ཡིན་དེས། །ཚུལ་འདི་ལ་ནི་བསླབ་པར་གྱིས། །བྱང་ཆུབ་དམ་པ་བལྟ་དཀའ་ཞིང་། །ཞི་བ་ཀུན་ཏུ་གཉེར་བྱེད་དང་། །འཁོར་ལོས་སྒྱུར་བ་ཐོབ་འདོད་པ། །དེས་ནི་ཚུལ་འདི་བསླབ་པར་གྱིས། །གཟུངས་ཀྱི་ཚུལ་འདི་བླ་མེད་ལ། །ཡང་དག་མཁས་པར་འདོད་ན་ཡང་། །ཆོས་འདི་ལ་ནི་ལེགས་བསླབས་ནས། །རྟག་ཏུ་བག་དང་ལྡན་པར་གྱིས། །སྨོན་ལམ་དམ་པ་རྒྱ་ཆེ་ཞིང་། །གཙོ་བོར་རྟག་ཏུ་ཐོབ་གྱུར་ནས། །བྱང་ཆུབ་རྟོགས་པར་འདོད་ན་ཡང་། །ཆོས་འདི་དག་ལ་བསླབ་པར་གྱིས། །མདོ་འདི་ལས་ནི་ངེས་སྨོས་པ། །ཕྱག་རྒྱ་བླ་ན་མེད་པ་སྟེ། །ཆོས་ཀུན་རྗོད་པར་བྱེད་པ་ཡི། །གཟུངས་སུའང་བཤད་པ་འདི་ཡིན་ནོ། །ཆོས་རྣམས་ཐམས་ཅད་གཟུངས་འདི་ཡི། །ནང་དུ་འདུས་པར་རབ་ཏུ་བཤད། །འདི་དག་ནམ་མཁའི་སྒོ་ཅན་ཏེ། །མཐའ་ཡས་དོན་དུ་ལེགས་པར་དེས། །ཆོས་འདི་དག་ལས་སྒོ་བསྟན་པ། །སྒོ་ནི་ལེགས་པར་བརྗོད་པ་ཡིན། །མཁས་པ་ཀུན་ཏུ་གཟུང་བའི་ཕྱིར། །གཟུངས་ནི་ཉེ་བར་སྐྱེ་བར་འགྱུར། །ཤེས་རབ་ངེས་པར་བསྟན་པ་འདི། །ཆོས་ཀུན་འཛིན་པར་བྱེད་པ་སྟེ། །མཁས་པ་ཀུན་ཏུ་གཟུང་བའི་ཕྱིར། །ཤེས་རབ་ཉེ་བར་གཟུང་བར་བྱ། །རྣམ་གྲངས་འདིར་ནི་བསྟན་པ་དག །སངས་རྒྱས་ཆོས་ཤེས་ཡང་དག་བྱ། །བྱང་ཆུབ་ཡན་ལག་བླ་ན་མེད། །དོན་གྱིས་ཡང་དག་བརྗོད་པ་ཡིན། །ཡེ་ཤེས་རབ་དབྱེ་མཁས་པ་ཡང་། །འདི་ལས་ཡང་དག་བསྟན་པ་ཡིན། །བྱང་ཆུབ་ཕྱོགས་དང་འཐུན་
པའི་ཆོས། །བླ་མེད་འདི་ལ་བསླབ་པར་གྱིས། །བླ་མེད་ཚུལ་ལ་མཁས་པ་ཡང་། །ཆོས་འདི་ལས་ནི་ངེས་པར་བསྟན། །ཐབས་ཤེས་པ་ལ་བརྟེན་ནས་ནི། །ཆོས་འདི་རབ་ཏུ་བསྟན་པར་གྱིས། །ཆོས་རྣམས་ཀྱི་ནི་རང་བཞིན་འདི། །ཉེ་བར་བསྟན་པ་བླ་ན་མེད། །འདི་ལ་བསླབས་ནས་བདུད་རྩི་ཡི། །དོན་འདི་ཡང་ནི་བཤད་པར་གྱིས། །མཁས་པ་གང་ཞིག་སངས་རྒྱས་ཀྱི། །ཡེ་ཤེས་ཆགས་པ་མེད་འདོད་པ། །དེ་དག་དོན་འདི་ལ་བསླབས་ན། །ཡེ་ཤེས་བླ་མེད་ཐོབ་པར་འགྱུར། །ངས་ཀྱང་སྔོན་ཆད་ཆོས་འདི་ལ། །བསྐལ་པ་བྱེ་བ་ཚད་མེད་ཅིང་། །གྲངས་མེད་དག་ཏུ་བསླབས་ནས་ནི། །ཚུལ་འདི་ཞི་བར་གྱུར་པ་ཡིན། །ངས་ནི་སྔོན་ཆད་སངས་རྒྱས་དག །སྟོང་ཕྲག་མང་པོ་མཉེས་པར་བྱས། །དེ་ཕྱིར་བླ་ན་མེད་པའི་ཆོས། །འདི་ཀུན་མཁྱེན་ཅིང་གཟིགས་པ་ཡིན། །ངས་ནི་ཁྱོད་དང་སེམས་ཅན་གྱི། །ཚད་གཟུང་མེད་ཅིང་མཐའ་ཡས་པའི། །དོན་ཆེན་བྱས་ཏེ་དེ་ལས་ནི། །གཟུངས་འདི་ཐོབ་པར་གྱུར་པ་ཡིན། །མཁས་པ་གང་ཞིག་གཟུངས་ཀྱི་སྒོའི། །ཕྱག་རྒྱ་འདི་ལས་ཚིག་གཅིག་ཅིག །རྣམ་པར་བསྒོམས་ནའང་ཚུལ་འདི་ལ། །རབ་ཏུ་འཇུག་པར་འགྱུར་བ་ཡིན། །ང་ཡི་ཡེ་ཤེས་ཚད་མེད་ཅིང་། །མཐའ་ཡས་བླ་ན་མེད་པ་སྟེ། །ང་ཡི་ཡེ་ཤེས་ཕུང་པོ་དེས། །ཆོས་འདི་རབ་ཏུ་བཤད་པར་མཛད། །ཤེས་རབ་ལྡན་པ་གང་ཞིག་ནི། །བྱང་ཆུབ་ཆེད་དུ་རྟོགས་འདོད་པ། །འཇིགས་མེད་དེ་ཡིས་དོན་ཚུལ་འདིར། །ཡང་དག་ཉིད་དུ་བརྩོན་པར་གྱིས། །མཁས་པ་གང་ཞིག་ཤེས་རབ་ཀྱི། །སྣང་བ་མང་པོ་འདོད་པ་དེས། །སངས་རྒྱས་བསྙེན་བཀུར་རབ་བྱས་ནས། །ཚུལ་འདི་དག་ལ་བསླབ་པར་གྱིས། །ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བ་དང་། །ཆོས་དུང་བུད་པར་འདོད་པ་གང་། །ཤེས་རབ་ལྡན་པ་དེ་ཡིས་ནི། །བསྟན་པ་འདི་ལ་བསླབ་པར་གྱིས། །མི་གང་འོད་ཟེར་མཐའ་ཡས་པ། །དགྱེ་བར་འདོད་ཅིང་སངས་རྒྱས་ཀྱི། ཆོས་དོན་གཉེར་བ་གང་ཡིན་དེས། །འདི་ལ་ཚུལ་བཞིན་བསླབ་པར་གྱིས། །དམ་པ་གང་ཞིག་ལྷར་བཅས་པའི། །འཇིག་རྟེན་ཟིལ་གནོན་འདོད་པ་དེས། །ཆོས་ཀུན་རྣམ་པར་ངེས་པ་ཡི། །མདོ་འདི་ལ་ནི་བསླབ་པར་གྱིས། །ང་ཡི་ཡེ་ཤེས་གཞལ་དུ་མེད། །ཡེ་ཤེས་རྩོམ་པའང་ཚད་མེད་དེ། །སངས་རྒྱས་ཡེ་ཤེས་འདོད་པ་གང་། །དེ་ཡིས་འདི་ལ་རྗེས་སུ་སློབས། །སངས་རྒྱས་ཡེ་ཤེས་སྤྲོས་མེད་ཅིང་། །ཡེ་ཤེས་དམ་པར་གྱུར་པའི་ཚུལ། །ཐམས་ཅད་རབ་ཏུ་སྟོན་འདོད་གང་། །དེ་ཡིས་དོན་འདི་བསླབ་པར་གྱིས། །གང་ཞིག་ཡེ་ཤེས་ཆགས་མེད་པ། །སྟོན་པ་རབ་ཏུ་འཆད་འདོད་པ། །དེ་དག་གིས་ནི་ཆོས་འདི་ལ། །བསླབས་ནས་བདུད་རྩི་བསྟན་པར་གྱིས། །གང་ཞིག་གིས་ནི་འཇིག་རྟེན་ཁམས། །བྱེ་བྲག་ཚད་མེད་མཐའ་ཡས་པ། །སྣང་བར་བྱེད་པར་འདོད་པ་དེས། །བསྟན་པ་འདི་ལ་ཡང་དག་སློབས། །སྒོ་འདི་བླ་ན་མེད་པ་སྟེ། །ཆོས་ཀུན་རྣམ་པར་སྦྱོང་བ་ཡིན། །མདོ་འདི་ལས་ནི་ཆོས་ཀུན་གྱི། །རྣམ་པར་དག་པ་རབ་ཏུ་བསྟན། །ཀུན་མཁྱེན་རྐང་གཉིས་མཆོག་གིས་ནི། །བླ་ན་མེད་པའི་མདོ་འདི་ལས། ཤིན་ཏུ་རྒྱས་པ་འདིར་བཤད་པ། །བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་བསྟན། །བཀོད་པ་མཐའ་ཡས་དེ་ལྟ་བས་ན་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་སེམས་ཅན་རྣམས་ལ་ཕན་གདགས་པའི་
